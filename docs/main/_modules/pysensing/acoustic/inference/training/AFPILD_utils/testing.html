[docs]def afpild_testing(
    config_file=".pysensing/acoustic/inference/training/AFPILD_utlis/configs/afpild_spec_gcc_fusion.json",
    root_dir="./",
    dataset_dir = "./data/audio_feature",
    resume_path='saved/AFPILD-CRNN/accil_ori_cloth/model/model_best.pth',
    device=None,
    mode='test',
    run_id='Fusion Model',
    log_name=None,
    task='accil_ori_cloth',

):

    """
    Function to run the test of the model.

    This function executes the testing phase for a machine learning model for AFPILD
    
    Args:
        config_file (str): Path to the configuration file containing model and test parameters.
        root_dir (str): Root directory for model and result saving.
        dataset_dir (str): Directory containing the dataset.
        resume_path (str): Path to the model checkpoint to resume testing from.
        device (str): Device to run the model on (e.g., 'cpu', 'cuda').
        mode (str): Mode of operation. Options: ["train", "test"]. (Default: "test")
        run_id (str): Unique identifier for the current test run.
        log_name (str): Name for the log file to record test results and metrics.
        task (str): Specific task or subset of the dataset to run the test on. Slected from: {ide/loc/ideloc/accil}_{ori/ana}_{rd/cloth/shoe}

    Returns:
        None

    Note:
        This function is based on AFPILD official repo: https://github.com/Cratial/AFPILD-CRNN

    Example:
        >>> from pysensing.acoustic.inference.training.AFPILD_utils.testing import afpild_testing
        >>> afpild_testing(
            config_file="./data/AFPILD/afpild_spec_gcc_fusion.json",
            root_dir= "./data/AFPILD",
            dataset_dir="./data/AFPILD/",
            resume_path="./data/AFPILD/saved/AFPILD-CRNN/20241030055348/model/model_best.pth", # Path to the trained model
            task='accil_ori_rd',
        )
    """

    # Create a parser for configuration
    parser = argparse.ArgumentParser(description='testing')
    run_args = parser.add_argument_group('run_args')
    run_args.add_argument('-c', '--config', default=config_file, type=str)
    run_args.add_argument('-r', '--resume', default=resume_path, type=str)
    run_args.add_argument('-d', '--device', default=device, type=str)
    run_args.add_argument('--mode', default=mode, type=str)
    run_args.add_argument('--run_id', default=run_id, type=str)
    run_args.add_argument('--log_name', default=log_name, type=str)
    run_args.add_argument('-t', '--task', default=task, type=str)

    mod_args = parser.add_argument_group('mod_args')
    CustomArgs = collections.namedtuple('CustomArgs', "flags default type target")
    options = [
        CustomArgs(['--root_dir'], type=str, target="root_dir", default="./"),
        CustomArgs(['--dataset_dir'], type=str, target=["datasets;train;data;kwargs;dataset_dir","datasets;valid;data;kwargs;dataset_dir","datasets;test;data;kwargs;dataset_dir"],default="./data/audio_feature")
    ]        

    for opt in options:
        mod_args.add_argument(*opt.flags, default=None, type=opt.type)

    # Build the list of command-line arguments
    cli_args = [
        '--config', config_file,
        '--resume', resume_path,
        '--device', device,
        '--mode', mode,
        '--run_id', run_id,
        '--task', task,
        "--root_dir", root_dir,
        "--dataset_dir", dataset_dir
    ]
    # if root_dir is not None:
    #     cli_args.extend(['--root_dir', str(root_dir)])

    # if dataset_dir is not None:
    #     cli_args.extend(['--dataset_dir', str(dataset_dir)])

    # if log_name is not None:
    #     cli_args.extend(['--log_name', log_name])

    # Initialize ConfigParser from parsed arguments
    config = ConfigParser.from_args(parser, options, args_list=cli_args)
    logger = get_logger('test')
    test_msg = msg_box("TEST: " + config.run_args.task)
    logger.debug(test_msg)
    # print(config.root_dir)
    # Load test datasets
    test_datasets = {}
    keys = ['datasets', 'test']
    for name in get_by_path(config, keys):
        test_datasets[name] = config.init_obj([*keys, name], 'data_loaders')

    # Data loaders
    test_data_loaders = {}
    keys = ['data_loaders', 'test']
    for name in get_by_path(config, keys):
        dataset = test_datasets[name]
        test_data_loaders[name] = config.init_obj([*keys, name], 'data_loaders', dataset)

    # Prepare model for testing
    device, device_ids = prepare_device(config['n_gpu'])
    # Load models
    logger.info(f"Loading model from {resume_path} ...")
    checkpoint = torch.load(resume_path)
    models_afpild = {}
    logger_model = get_logger('model', verbosity=0)
    for name in config['models_afpild']:
        model = config.init_obj(['models_afpild', name], 'models_afpild')
        logger_model.info(model)
        state_dict = checkpoint['models'][name]
        if config['n_gpu'] > 1:
            model = torch.nn.DataParallel(model)
        model.load_state_dict(state_dict)
        model = model.to(device)
        model.eval()
        models_afpild[name] = model

    # Losses
    losses = {}
    for name in config['losses']:
        kwargs = {}
        losses[name] = config.init_obj(['losses', name], module_loss, **kwargs)

    # Metrics
    metrics_iter = [getattr(module_metric, met) for met in config['metrics']['per_iteration']]
    metrics_epoch = [getattr(module_metric, met) for met in config['metrics']['per_epoch']]
    keys_loss = ['loss']
    keys_iter = [m.__name__ for m in metrics_iter]
    keys_epoch = [m.__name__ for m in metrics_epoch]
    test_metrics = MetricTracker(keys_loss + keys_iter, keys_epoch)

    task = config.run_args.task.split('_')[0]

    with torch.no_grad():
        print("Testing...")
        model = models_afpild['model']
        testloader = test_data_loaders['data']
        if len(metrics_epoch) > 0:
            outputs = torch.FloatTensor().to(device)
            targets = torch.FloatTensor().to(device)

        for batch_idx, (data, target) in tqdm(enumerate(testloader), total=len(testloader)):
            if isinstance(data, dict):
                data = {k: v.to(device).float() for k, v in data.items()}
            else:
                data = data.to(device).float()

            target = target.to(device).float()
            output = model(data)

            if len(metrics_epoch) > 0:
                outputs = torch.cat((outputs, output))
                targets = torch.cat((targets, target))

            # Computing loss and metrics on the test set
            if task == 'ide':
                loss = losses['ide_loss'](output, target.type(torch.int64))
            elif task == 'loc':
                loss = losses['loc_loss'](output, target)
            elif task == 'ideloc':
                ide_loss = losses['ide_loss'](output[:, :40], target[:, 0].type(torch.int64))
                loc_loss = losses['loc_loss'](output[:, 40], target[:, 1])
                loss = ide_loss + loc_loss
            elif task == 'accil':
                loss = losses['accdoa_loss'](output, target)

            test_metrics.iter_update('loss', loss.item())
            for met in metrics_iter:
                test_metrics.iter_update(met.__name__, met(output, target))

        for met in metrics_epoch:
            test_metrics.epoch_update(met.__name__, met(outputs, targets))

    print(outputs.cpu().numpy().shape)

    # Save prediction and targets
    save_dir = config.save_dir["metric"]
    np.savez(os.path.join(save_dir, 'pred_target'), pred=outputs.cpu().numpy(), target=targets.cpu().numpy())

    # Log test results
    test_log = test_metrics.result()
    logger.info(test_log)
