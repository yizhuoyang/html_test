[docs]def train_model(
    config_path: str,
    resume: bool = False,
    seed: int = 0,
) -> None:
    """
    Function to train the acoustic hpe mpdel.

    Args:
        config_path (str): Path to the configuration file containing model and execution parameters.
        resume (bool): Whether to use the pretrain model for initialization
        seed (int): set the random seed

    Returns:
        None

    Note:
        The code is based on the officail repo of AcousticPose: https://github.com/YutoShibata07/AcousticPose_Public

    Example:
        >>> from pysensing.acoustic.inference.training.AcousticPose_utils.hpe_train import train_model, generate_configs
        >>> args = {
        ...    "root_dir": "./data/hpe_dataset/testing_result",  # Directory to save training results
        ...    "save_name": "seq1",  # Experiment name or identifier for saving files
        ...    "input_feature": "logmel",  # Type of input features (e.g., 'logmel')
        ...    "batchsize": 64,  # Training batch size
        ...    "max_epoch": 50,  # Maximum number of epochs to train
        ...    "csv_path": "./data/hpe_dataset/csv",  # Path to the dataset in CSV format
        ...    "dataset_name": "pose_regression_timeseries_subject_1",  # Name of the dataset to use
        ...    "model": "speech2pose",  # Model name or type
        ...    "sound_length": 2400,  # Length of input sound sequences
        ...    "learning_rate": 0.01,  # Initial learning rate for the optimizer
        }
        >>> config_path = args["root_dir"] + '/' + args["save_name"] + "/config.yaml"
        >>> generate_configs(args)
        >>> resume_training = False  # Whether to resume training from a checkpoint
        >>> random_seed = 0  # Seed for reproducibility
        >>> train_model(
        ...    config_path=config_path,  # Path to configuration YAML file
        ...    resume=resume_training,   # Boolean flag to resume or start fresh training
        ...    seed=random_seed,         # Random seed for ensuring reproducibility
        )
    """
    result_path = os.path.dirname(config_path)
    experiment_name = os.path.basename(result_path)

    logname = os.path.join(result_path, f"{datetime.datetime.now():%Y-%m-%d}_train.log")
    basicConfig(
        level=INFO,
        format="[%(asctime)s] %(name)s %(levelname)s: %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S",
        filename=logname,
    )

    set_seed(seed)

    config = get_config(config_path)

    device = get_device(allow_only_gpu=False)

    train_loader = get_dataloader_temp(
        config.csv_path,
        config.dataset_name,
        config.sound_length,
        config.input_feature,
        "train",
        batch_size=config.batch_size,
        shuffle=True,
        num_workers=config.num_workers,
        pin_memory=True,
        drop_last=True,
    )

    val_loader = get_dataloader_temp(
        config.csv_path,
        config.dataset_name,
        config.sound_length,
        config.input_feature,
        "val",
        batch_size=1,
        shuffle=False,
        num_workers=config.num_workers,
        pin_memory=True,
    )


    n_classes = len(get_cls2id_map())

    model = get_model(
        config.model,
        n_classes,
        config.input_feature,
        pretrained=config.pretrained,
    )

    model.to(device)
    optimizer = optim.Adam(model.parameters(), lr=config.learning_rate)

    begin_epoch = 0
    best_rmse = float("inf")

    if resume:
        resume_path = os.path.join(result_path, "checkpoint.pth")
        begin_epoch, model, optimizer, best_rmse = resume_checkpoint(resume_path, model, optimizer)

    log_path = os.path.join(result_path, "log.csv")

    criterion = get_criterion(config.ratio)

    logger.info("Start training.")
    train_logger = TrainLogger(log_path, resume=resume)

    for epoch in range(begin_epoch, config.max_epoch):
        start = time.time()
        train_loss, train_rmse, train_mae, train_acc = train(
            train_loader, model, criterion, optimizer, epoch, device, aug=config.aug, smooth_loss=config.smooth_loss
        )
        train_time = int(time.time() - start)

        start = time.time()
        val_loss, val_rmse, val_mae, val_acc = evaluate(
            val_loader, model, criterion, device
        )
        val_time = int(time.time() - start)
        train_logger.update(
            epoch,
            optimizer.param_groups[0]["lr"],
            train_time,
            train_loss,
            train_rmse,
            train_mae,
            train_acc,
            val_time,
            val_loss,
            val_rmse,
            val_mae,
            val_acc,
        )
        if best_rmse > val_rmse["all"]:
            best_rmse = val_rmse["all"]
            torch.save(
                model.state_dict(),
                os.path.join(result_path, "best_model.prm"),
            )

        save_checkpoint(result_path, epoch, model, optimizer, best_rmse)

        print(
            f"epoch: {epoch}\tepoch time[sec]: {train_time + val_time}\t"
            f"train loss: {train_loss:.4f}\tval loss: {val_loss:.4f}\t"
            f"val_rmse: {val_rmse['all']:.5f}\tval_mae: {val_mae['all']:.5f}"
        )

    torch.save(model.state_dict(), os.path.join(result_path, "final_model.prm"))

    os.remove(os.path.join(result_path, "checkpoint.pth"))

    logger.info("Done")
