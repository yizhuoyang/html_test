[docs]class timestretch(torch.nn.Module):
    r"""
    Data augmentation method by stretching the time scale of the input STFT

    Args:
        hop_length (int or None, optional): length of hop between STFT windows.
        n_freq (int, optional): number of filter banks from stft.
        fixed_rate (float or None, optional): rate to speed up or slow down by.

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> timestretch_trans  = A.spectrogram_aug.timestretch()
        >>> timestretch_spec   = timestretch_trans(spectrogram)

    Note:
        The implementation is based on torchaudio.transforms.TimeStretch.

    Reference: 
        https://pytorch.org/audio/main/_modules/torchaudio/transforms/_transforms.html#TimeStretch
    """
    def __init__(
        self, 
        hop_length: Optional[int] = None, 
        n_freq: int = 201, 
        fixed_rate: Optional[float] = None
        ) -> None:
            
        super(timestretch, self).__init__()
        self.transformation = torchaudio.transforms.TimeStretch(
            hop_length = hop_length,
            n_freq = n_freq,
            fixed_rate = fixed_rate
        )


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            complex_specgrams (Tensor):
                A tensor of dimension `(..., freq, num_frame)` with complex dtype.
            overriding_rate (float or None, optional): speed up to apply to this batch.

        Returns:
            Tensor:
                The stretched spectrogram.
        """
        timestretch_result = self.transformation(waveform)

        return timestretch_result

    


[docs]class timemasking(torch.nn.Module):
    r"""
    Data augmentation method by masking the time scale of the input STFT.

    Args:
        time_mask_param (int): The maximum possible length of the mask,
            with indices sampled uniformly from the range [0, time_mask_param).
        iid_masks (bool, optional): Indicates whether different masks should be applied to each example/channel in the batch. 
            Only applicable when the input tensor has 3 or more dimensions. Default is ``False``.
        p (float, optional): The maximum proportion of time steps that can be masked. 
            Must be a value within the range [0.0, 1.0]. Default is 1.0.

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> timemasking_trans  = A.spectrogram_aug.timemasking(time_mask_param=100)
        >>> timemasking_spec   = timemasking_trans(spectrogram)

    Note:
        The implentation is based on torchaudio.transforms.TimeMasking.

    Reference: 
        https://pytorch.org/audio/main/generated/torchaudio.transforms.TimeMasking.html#torchaudio.transforms.TimeMasking
    """
    def __init__(
        self, 
        time_mask_param: int, 
        iid_masks: bool = False, 
        p: float = 1.0
        ) -> None:
            
        super(timemasking, self).__init__()
        self.transformation = torchaudio.transforms.TimeMasking(
            time_mask_param = time_mask_param,
            iid_masks = iid_masks,
            p = p
        )


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            specgram (Tensor): Tensor of shape `(..., freq, time)`.
        Returns:
            Tensor: Masked spectrogram of shape `(..., freq, time)`.
        """
        timemasking_result = self.transformation(waveform)

        return timemasking_result

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> add_noise_trans   = A.signal_aug.add_noise(n_fft=400)
        >>> noise_data        = add_noise_trans(waveform)

    Note:
        The implementation is based on torchaudio.transforms.FrequencyMasking.    

    Reference: 
        https://pytorch.org/audio/main/generated/torchaudio.transforms.FrequencyMasking.html#torchaudio.transforms.FrequencyMasking
    """
    def __init__(
        self, 
        freq_mask_param: int, 
        iid_masks: bool = False
        ) -> None:
            
        super(frequencymasking, self).__init__()
        self.transformation = torchaudio.transforms.FrequencyMasking(
            freq_mask_param = freq_mask_param,
            iid_masks = iid_masks
        )
