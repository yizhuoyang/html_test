[docs]def get_neighbors(neighbors_array, point_idx: int, level: int) -> np.ndarray:
    r'''
    Retrieves the neighbors of a given point from a mesh for a specified level.

    Args:
        neighbors_array: Array containing the neighbors of each point in the mesh.
                         Shape (N, k), where N is the number of vertices and k is the number of neighbors per vertex.
        point_idx (int): Index of the point whose neighbors are to be found.
        level (int): Level of neighbors to consider. Level 1 returns direct neighbors, level 2 returns neighbors' neighbors, etc.

    Returns:
        Array of indices representing the neighbors of the point at the specified level.
    '''
    neighbors = np.asarray(neighbors_array[point_idx])
    for _ in range(level - 1):
        cur_level_neighbors = []
        for neighbor in neighbors:
            cur_level_neighbors.extend(neighbors_array[neighbor])
        neighbors = np.asarray(list(set(cur_level_neighbors)))
    return neighbors




[docs]def get_vertices_defined_curvature(mesh: trimesh.Trimesh, sampled_indices: List[int], level: int = 2) -> np.ndarray:
    r'''
    Calculates the average angle curvature at each sampled vertex using neighboring vertices' normals.

    Args:
        mesh (trimesh.Trimesh): The input mesh object containing vertex data and neighbors.
        sampled_indices (List[int]): Array of vertex indices to sample and calculate curvature for.
        level (int, optional): The level of neighbors to consider for calculating curvature. Default is 2.

    Returns:
        Array of shape (sample_size, 1) representing the average angle (curvature) for each sampled vertex.
    '''
    normals = mesh.vertex_normals  # N, 3
    neighbors_array = mesh.vertex_neighbors # N, k

    # iterate each point, get its N level neighbors, calculate the average normal angles.
    average_angle = []
    for point_idx in sampled_indices:
        neighbors = get_neighbors(neighbors_array, point_idx, level)  # k,
        # calcuate the angles
        neighbors_normals = normals[neighbors, :]  # k,3
        cosine_array = np.dot(normals[point_idx].reshape(1, -1), neighbors_normals.T)  # 1,k
        angle_array = np.arccos(np.clip(cosine_array, -1.0, 1.0))  # 1,k
        average_angle.append(np.mean(angle_array))
    average_angle = np.reshape(average_angle, (len(average_angle), 1))  # N,1
    return average_angle




[docs]def extract_feature_labels(obj_path: str, gt_path: str, sample_size: int) -> Tuple[np.ndarray, np.ndarray]:
    r'''
    Extracts features and labels for a given 3D object file and its corresponding ground truth labels.

    Args:
        obj_path (str): Path to the .obj file containing mesh data.
        gt_path (str): Path to the .json file containing FDI GT labels.
        sample_size (int): Number of points to sample from the mesh. If sample_size is 0, all points are used.

    Returns:
        Tuple(np.ndarray, np.ndarray): 
            - obj_feas (np.ndarray): Array of shape (sample_size, 8) containing the features of the sampled points.
            - label (np.ndarray): Array of shape (sample_size,) containing the corresponding labels for the sampled points.

    Example:
        >>> obj_path = "path/to/mesh.obj"
        >>> gt_path = "path/to/labels.json"
        >>> sample_size = 500
        >>> features, labels = extract_feature_labels(obj_path, gt_path, sample_size)
        >>> print(features.shape)  # Should print: (500, 8)
        >>> print(labels.shape)  # Should print: (500,)
    '''
    with open(obj_path) as F:
        teeth_dict = trimesh.exchange.obj.load_obj(F)
    points_obj = np.array(teeth_dict['vertices'])
    with open(gt_path, "r") as F:
        label = json.load(F)['labels']
    label = np.array(label, dtype=np.int64)
    assert points_obj.shape[0] == len(label), 'mesh and label does not match'

    np.random.seed(1024)
    if sample_size != 0:
        sampled_indices = np.random.choice(np.arange(len(label)), replace=False, size=sample_size)
    else:
        sampled_indices = np.arange(len(label))

    raw_mesh = trimesh.exchange.load.load_kwargs(teeth_dict)    # Load mesh
    points_mesh_orig = raw_mesh.vertices        # Vertices of the mesh (in different order from points_obj)
    vertex_normals = raw_mesh.vertex_normals    # Normals for each vertex, shape (N, 3)
    
    # Find sampled points and their corresponding mesh indices
    points_mesh_sampled, mesh_sample_indices = [], []
    for idx, pts in enumerate(points_mesh_orig):
        if pts in points_obj[sampled_indices]:
            points_mesh_sampled.append(tuple(pts))
            mesh_sample_indices.append(idx)

    # Calculate curvature features for the sampled points
    gaussian_curvature = trimesh.curvature.discrete_gaussian_curvature_measure(raw_mesh, points_mesh_orig[mesh_sample_indices], 0)
    gaussian_curvature = np.reshape(gaussian_curvature, (len(gaussian_curvature), 1))  # sample_size,
    AvgAngle_curvature = get_vertices_defined_curvature(raw_mesh, sampled_indices=mesh_sample_indices)  # sample_size,
    
    # Concatenate position, normal, and curvature features for each sampled point
    pts_normals_curvatures = np.hstack((points_mesh_orig[mesh_sample_indices], vertex_normals[mesh_sample_indices],
                                        gaussian_curvature, AvgAngle_curvature))  # sample_size, 8

    # Map sampled points to their corresponding features
    obj_fea_list = []
    point_fea_mapping = dict(zip(points_mesh_sampled, pts_normals_curvatures))
    for pts in points_obj[sampled_indices]:
        obj_fea_list.append(point_fea_mapping[tuple(pts)])
    obj_feas = np.stack(obj_fea_list, axis=0)  # sample_size, 8

    return obj_feas, label[sampled_indices]




[docs]def process_part_folders(part_raw_folders: list, save_path: str, sample_size: int):
    r'''
    Processes raw folders containing 3D object data and saves the extracted features and labels.

    Args:
        part_raw_folders (list): List of folder paths containing the raw .obj and .json files.
        save_path (str): Path to save the processed features and labels in JSON format.
        sample_size (int): Number of points to sample from each mesh.
    '''
    for case_path in tqdm(part_raw_folders, total=len(part_raw_folders)):
        if os.path.isdir(case_path):
            case = os.path.split(case_path)[1]
            category = os.path.basename(os.path.split(case_path)[0])

            if os.path.exists(os.path.join(save_path, category, case, '.json')):
                continue

            all_obj_paths = glob.glob(os.path.join(case_path, '*.obj'))
            gt_path = os.path.join(case_path, case + '_' + category + '.json')

            if not os.path.exists(os.path.join(save_path, category, case)):
                os.makedirs(os.path.join(save_path, category, case))
            for obj_path in all_obj_paths:
                fea, lbl = extract_feature_labels(obj_path, gt_path, sample_size)  # (SAMPLE_SIZE, 8), (SAMPLE_SIZE,)
                lbl = np.array([FDI_cls_mapping[x] for x in lbl], dtype=int)

                save_case_path = os.path.join(save_path, category, case, os.path.basename(obj_path)[:-3]+'json')
                with open(save_case_path, 'w') as F:
                    json.dump({'feature': fea.tolist(),  # (SAMPLE_SIZE, 8)
                               'label': lbl.tolist(),  # (SAMPLE_SIZE,)
                               'category': [1.0, 0.0] if category == 'lower' else [0.0, 1.0]}, F)  # (2,)




[docs]def multithreading_pre(root_path: str, save_path: str, workers: int = 4, sample_size: int = 10000):
    r'''
    Processes multiple parts of 3D mesh datasets concurrently using multithreading.

    Args:
        root_path (str): Root directory containing all the data for the segmentation.
        save_path (str): Path to save processed data.
        workers (int, optional): Number of threads to use for concurrent processing. Default is 4.
        sample_size (int, optional): Number of points to sample from each mesh. Default is 10000.
    '''
    data_paths = []
    for jaw_dir in os.listdir(root_path):
        jaw_path = os.path.join(root_path, jaw_dir)
        if os.path.isdir(jaw_path):
            for case in os.listdir(jaw_path):
                data_paths.append(os.path.join(jaw_path, case))
    
    ths, unit = [], math.ceil(len(data_paths)/workers)
    for order in np.arange(workers):
        data_part_paths = data_paths[order * unit: min((order + 1) * unit, len(data_paths))]
        th = threading.Thread(target=process_part_folders, args=(data_part_paths, save_path, sample_size))
        th.start()
        ths.append(th)
    for th in ths:
        th.join()
