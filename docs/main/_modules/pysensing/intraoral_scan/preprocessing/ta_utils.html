# Define the mapping between the line_id of pose in axis file and FDI numbering system.
fdi_seq = [i for i in range(17, 10, -1)] \
    + [i for i in range(21, 28)] \
    + [i for i in range(47, 40, -1)] \
    + [i for i in range(31, 38)]
fdi2id_dict = {fdi: id for id, fdi in enumerate(fdi_seq)}



[docs]def get_idx_dirs(root: str) -> Tuple[List[int], List[str]]:
    r'''
    Reads the index and directory mappings from a file.

    Args:
        root (str): The root directory where the `idx_dirs.txt` file is located.

    Returns:
        Tuple (list[int], list[str]): 
            - idx (list[int]): List of indices extracted from the file.
            - dirs (list[str]): List of directory names corresponding to each index.

    Example:
        >>> root = "path/to/tooth_alignment_example"
        >>> idx, dirs = get_idx_dirs(root)
        >>> print(idx, dirs)
    '''
    idx, dirs = [], []
    with open(f"{root}/idx_dirs.txt", 'r') as file:
        for line in file:
            idx.append(line.strip().split()[0])
            dirs.append(line.strip().split()[1])
    return idx, dirs




[docs]def getAxis(path: str, keep_fdi: bool = False) -> torch.Tensor:
    r'''
    Loads axis information from a file and returns it as a tensor.

    Args:
        path (str): Path to the file containing the axis information.
        keep_fdi (bool): If True, the FDI tooth number will be included in the tensor. 
                         If False, only the quaternion and positional data will be included.

    Returns:
        A tensor of shape (28, 8) if `keep_fdi=True`, otherwise (28, 7).
        - If `keep_fdi=True`, the tensor includes FDI numbers followed by axis data.
        - If `keep_fdi=False`, the tensor contains only the axis data (quaternion and position).

    Example:
        >>> axis = getAxis("path/to/TeethAxis_Ori.txt", keep_fdi=True)
        >>> print(axis.shape)  # Output: torch.Size([28, 8])
    '''
    axis = torch.zeros((28, (8 if keep_fdi==True else 7)), dtype=torch.float32)
    with open(path, 'r') as file:
        for _ in range(28):
            cur_line = file.readline().strip().split(' ')
            fdi, pose = int(cur_line[0]), cur_line[1:]
            if keep_fdi == True:
                axis[fdi2id_dict[fdi]] = torch.tensor([float(fdi)] + [float(x) for x in pose], dtype=torch.float32)
            else:
                axis[fdi2id_dict[fdi]] = torch.tensor([float(x) for x in pose], dtype=torch.float32)
    return axis




[docs]def QuatMul(p: torch.Tensor, q: torch.Tensor, invp: bool = False, invq: bool = False) -> torch.Tensor:
    r'''
    Performs quaternion multiplication between two quaternions `p` and `q`.

    Args:
        p (Tensor): The first quaternion tensor of shape (..., 4), where the last dimension represents (x, y, z, w).
        q (Tensor): The second quaternion tensor of shape (..., 4), where the last dimension represents (x, y, z, w).
        invp (bool): If True, the first quaternion `p` is inverted before multiplication. Default is False.
        invq (bool): If True, the second quaternion `q` is inverted before multiplication. Default is False.

    Returns:
        The resulting quaternion from the multiplication of `p` and `q` of shape (..., 4).
        
    Example:
        >>> p = torch.tensor([0., 0., 0., 1.])
        >>> q = torch.tensor([0., 1., 0., 0.])
        >>> res = QuatMul(p, q)
        >>> print(res)  # Output: tensor([0., 0., 0., -1.])
    '''
    pw, px, py, pz = p[..., 3], p[..., 0], p[..., 1], p[..., 2]
    qw, qx, qy, qz = q[..., 3], q[..., 0], q[..., 1], q[..., 2]
    if invp:
        px, py, pz = -px, -py, -pz
    if invq:
        qx, qy, qz = -qx, -qy, -qz
    res = torch.stack(
        [pw * qx + px * qw + py * qz - pz * qy,
        pw * qy + py * qw + pz * qx - px * qz,
        pw * qz + pz * qw + px * qy - py * qx,
        pw * qw - px * qx - py * qy - pz * qz],
        axis=-1)
    return res

    


[docs]def QuatRotate(x: torch.Tensor, q: torch.Tensor, inv: bool = False) -> torch.Tensor:
    r'''
    Rotates a vector `x` using a quaternion `q`.

    Args:
        x (Tensor): The input vector tensor of shape (..., 3), representing (x, y, z) coordinates.
        q (Tensor): The quaternion tensor of shape (..., 4), representing the rotation in (x, y, z, w) format.
        inv (bool): If True, the quaternion `q` is inverted to perform an inverse rotation. Default is False.

    Returns:
        The rotated vector of shape (..., 3).
        
    Example:
        >>> x = torch.tensor([1., 0., 0.])
        >>> q = torch.tensor([0., 0., 0., 1.])
        >>> rotated_x = QuatRotate(x, q)
        >>> print(rotated_x)  # Output: tensor([1., 0., 0.])
    '''
    invq = torch.concat([-q[..., :3], q[..., 3, None]], axis=-1)
    if inv:
        q, invq = invq, q
    x = F.pad(input=x, pad=(0, 1), mode='constant')
    x = QuatMul(q, x)
    x = QuatMul(x, invq)
    return x[..., :3]




[docs]def trans_pred(data_loader, preds: torch.Tensor, root: str) -> torch.Tensor:
    r'''
    Transforms predicted tooth pose data using original and target axis information.

    Args:
        data_loader: DataLoader object that provides coords of input data.
        preds (Tensor): Tensor of predicted pose of shape (batch_size, 28, 7),
                        where the first 4 values are quaternions and the last 3 are positions.
        root (str): The root directory where the axis data is stored.

    Returns:
        The transformed poses for each tooth in the batch, of shape (batch_size, 28, 8).
        Each row corresponds to [FDI, quaternion (4 values), position (3 values)] for 28 teeth.
    '''
    coords_all = []
    for coords, _ in data_loader.batches():
        coords_all += coords
    coords_all = torch.stack(coords_all)

    (_, dirs) = get_idx_dirs(root)
    ans_pose, preds = [], preds.cpu()
    for id in range(3):
        axis_o = getAxis(f'{root}/{dirs[id]}/TeethAxis_Ori.txt')
        axis_t = getAxis(f'{root}/{dirs[id]}/TeethAxis_T2.txt')

        quat = preds[id, :, :4]
        pos = preds[id, :, 4:]
        cur_coords = coords_all[id, :]

        cur_coords = cur_coords - axis_o[:, None, 4:]
        cur_coords = QuatRotate(cur_coords, axis_o[:, None, :4], inv=True)
        center = torch.mean(cur_coords, dim=1)

        quat = QuatMul(quat, axis_o[:, :4])
        pos = pos + axis_o[:, 4:] + QuatRotate(center, axis_o[:, :4], inv=False) - QuatRotate(center, axis_t[:, :4], inv=False)

        cur_patient_pose = torch.empty((28, 8), dtype=torch.float32)
        for i in range(28):
            cur_tooth_pose = [fdi_seq[i]]
            cur_tooth_pose += quat[i, :].tolist()
            cur_tooth_pose += pos[i, :].tolist()
            cur_patient_pose[i] = torch.tensor(cur_tooth_pose, dtype=torch.float32)
        ans_pose.append(cur_patient_pose)
    
    return torch.stack(ans_pose)




[docs]def quaternion_to_matrix(vector: torch.Tensor) -> torch.Tensor:
    r'''
    Converts a quaternion vector into a 4x4 transformation matrix.

    Args:
        vector (Tensor): A tensor of shape (7,) where the first 4 values represent a quaternion (x, y, z, w),
                         and the last 3 values represent a translation vector (tx, ty, tz).

    Returns:
        A 4x4 transformation matrix, where the upper-left 3x3 submatrix represents the rotation,
        and the last column represents the translation.

    Example:
        >>> vector = torch.tensor([0., 0., 0., 1., 1., 2., 3.])
        >>> M = quaternion_to_matrix(vector)
        >>> print(M)  # Output: 4x4 transformation matrix
    '''
    M = torch.eye(4)
    M[:3, 3] = vector[4:]
    r = R.from_quat(vector[:4])
    M[:3, :3] = torch.from_numpy(r.as_matrix())
    return M




[docs]def getTooth(patient_id: int, root: str, dirs: List[str]) -> List:
    r'''
    Loads 3D tooth models for a given patient from the corresponding directory.

    Args:
        patient_id (int): The ID of the patient to load tooth models for.
        root (str): The root directory where patient data is stored.
        dirs (list[str]): A list of directory names corresponding to different patients.

    Returns:
        A list of trimesh objects representing each tooth model for the patient.
    '''
    tooth_path = f'{root}/{dirs[patient_id]}/crown_tooth[0-9][0-9].stl'
    tooth = sorted(glob.glob(tooth_path))
    # print(tooth)
    objs = [trimesh.load(path) for path in tooth]
    return objs




[docs]def showTooth(objs: list, pose: torch.Tensor) -> trimesh.Trimesh:
    r'''
    Applies transformations to the tooth objects and merges them into a single 3D model.

    Args:
        objs (list): A list of trimesh objects representing the 3D tooth models.
        pose (Tensor): A tensor of shape (28, 8), where each row contains FDI, quaternion (4 values), 
                       and position (3 values) for each tooth.

    Returns:
        Trimesh: A merged 3D object of all teeth after transformation.

    Example:
        >>> objs = getTooth(0, root, dirs)
        >>> pose = torch.randn(28, 8)  # Random pose data
        >>> result_obj = showTooth(objs, pose)
        >>> result_obj.show()  # Visualize the merged 3D model
    '''
    specs = sorted(pose, key = lambda x: x[0])
    tran = torch.stack([quaternion_to_matrix(spec[1:]) for spec in specs])
    objs_tar = [obj.copy().apply_transform(t) for obj, t in zip(objs, tran)]
    result_obj = None
    for obj in objs_tar:
        result_obj += obj
    return result_obj
