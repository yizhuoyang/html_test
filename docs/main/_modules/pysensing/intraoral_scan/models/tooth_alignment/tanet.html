import torch
import torch.nn as nn
import torch.nn.functional as F



[docs]def custom_max(x, axis):
    r'''
    Performs max pooling along one or multiple dimensions.

    Args:
        x (Tensor): Input tensor.
        axis (int or list/tuple): Axis or axes along which to apply max pooling.

    Returns:
        Resulting tensor after max pooling along the specified axes.
    '''
    if isinstance(axis, tuple) or isinstance(axis, list):
        for ax in sorted(axis, reverse=True):
            x = torch.max(x, dim=ax)[0]
    else:
        x = torch.max(x, dim=axis)[0]
    return x




[docs]class PointNet(nn.Module):
    r'''
    A simple PointNet encoder that applies MLP layers to point cloud data and performs global max pooling.
    
    Args:
        units (list): A list of integers specifying the number of units for each layer.
        axis (int or tuple): Axis or axes along which to apply max pooling.
    '''
    def __init__(self, units=[64, 128, 1024], axis=-2):
        super(PointNet, self).__init__()
        self.convs = nn.ModuleList([nn.Linear(in_features, out_features) for in_features, out_features in zip([3] + units[:-1], units)])
        self.axis = axis
    

[docs]    def forward(self, x):
        r'''
        Forward pass through the PointNet encoder.

        Args:
            x (Tensor): Input point cloud tensor of shape (batch_size, npoints, dim, features).

        Returns:
            Output feature tensor after applying MLP and max pooling.
        '''
        for conv in self.convs:
            x = F.relu(conv(x))
        out = custom_max(x, self.axis)
        return out




[docs]class GRUCell(nn.Module):
    r'''
    Custom GRU cell implementation.
    
    Args:
        unit (int): The number of hidden units for the GRU cell.
    '''
    def __init__(self, unit):
        super(GRUCell, self).__init__()
        self.xdz = nn.Linear(unit, unit, bias=False)
        self.hdz = nn.Linear(unit, unit)
        self.xdr = nn.Linear(unit, unit, bias=False)
        self.hdr = nn.Linear(unit, unit)
        self.xdh = nn.Linear(unit, unit, bias=False)
        self.hdh = nn.Linear(unit, unit)


[docs]    def forward(self, inputs, states):
        r'''
        Forward pass through the GRU cell.

        Args:
            inputs (Tensor): Input tensor of shape (batch_size, unit).
            states (Tensor): Hidden state tensor of shape (batch_size, unit).

        Returns:
            Updated hidden state.
        '''
        z = torch.sigmoid(self.xdz(inputs) + self.hdz(states))
        r = torch.sigmoid(self.xdr(inputs) + self.hdr(states))
        hp = torch.tanh(self.xdh(inputs) + self.hdh(r * states))
        h = (1 - z) * states + z * hp
        return h

    


[docs]class TANet(nn.Module):
    r"""
    Implementation of "TANet: Towards Fully Automatic Tooth Arrangement".

    The model processes point cloud data representing tooth shapes and outputs pose transformations (quaternion and position).
    """
    def __init__(self, name='tanet'):
        super(TANet, self).__init__()
        self.global_encoder = PointNet([64, 128, 1024], axis=(1, 2))
        self.local_encoder = PointNet([64, 128, 512])

        self.adenses = nn.ModuleList([nn.Linear(512, 512, bias=False) for _ in range(3)])
        self.sdenses = nn.ModuleList([nn.Linear(512, 512, bias=False) for _ in range(3)])
        self.cdenses = nn.ModuleList([nn.Linear(512, 512, bias=False) for _ in range(3)])
        self.jdenses = nn.ModuleList([nn.Linear(512, 512, bias=False) for _ in range(3)])

        self.gru = GRUCell(512)

        self.adj = torch.zeros((28, 4), dtype=torch.int64)
        self.adj[1:14, 0] = torch.arange(13) + 1
        self.adj[15:28, 0] = torch.arange(13) + 15
        self.adj[0:13, 1] = torch.arange(13) + 2
        self.adj[14:27, 1] = torch.arange(13) + 16
        self.adj[0:14, 2] = torch.arange(13, -1, -1) + 29
        self.adj[14:28, 2] = torch.arange(13, -1, -1) + 43
        self.adj[0:14, 3] = 57
        self.adj[14:28, 3] = 58

        self.mlp = nn.ModuleList([nn.Linear(512 + 3 + 1024, 128), nn.Linear(128, 64), nn.Linear(64, 7)])
        nn.init.normal_(self.mlp[-1].weight, mean=0., std=1e-4)
    

[docs]    def forward(self, x):
        r'''
        Forward pass through TANet.

        Args:
            x (Tensor): Input tensor of shape (batch_size, ntooth, npoint, nfeature).

        Returns:
            Tuple(Tensor, Tensor):
                   - quat (Tensor): Predicted quaternions of shape (batch_size, ntooth, 4).
                   - pos (Tensor): Predicted positions of shape (batch_size, ntooth, 3).

        Example:
            >>> model = TANet()
            >>> x = torch.randn(4, 28, 400, 3)
            >>> quat, pos = model(x)
            >>> print(quat.shape, pos.shape)  # Output: (4, 28, 4), (4, 28, 3)
        '''
        batch_size = x.size(0)
        center = torch.mean(x, dim=2)
        gcode = self.global_encoder(x)

        x = x - center[:, :, None]
        lcode = self.local_encoder(x)
        htooth = lcode
        hjaw = torch.zeros((batch_size, 2, 512), dtype=torch.float32, device=x.device)
        h = torch.cat([htooth, hjaw], dim=1)
        zeros = torch.zeros((batch_size, 1, 512), dtype=torch.float32, device=x.device)

        for i in range(3):
            ha = self.adenses[i](htooth)
            hs = self.sdenses[i](htooth)
            hc = self.cdenses[i](htooth)
            hjc = self.cdenses[i](hjaw)
            hj = self.jdenses[i](hjaw)

            htooth = torch.cat([zeros, ha, hs, hjc], dim=1)
            htooth = torch.stack([torch.stack([htooth[:, i] for i in cur_adj], dim=1) for cur_adj in self.adj], dim=1)
            htooth = torch.sum(htooth, dim=2)
            hup = torch.sum(hc[:, :14], dim=1) + hj[:, 1]
            hdown = torch.sum(hc[:, 14:], dim=1) + hj[:, 0]

            x = torch.cat([htooth, hup[:, None], hdown[:, None]], dim=1)
            h = self.gru(x, h)
            htooth, hjaw = torch.split(h, [28, 2], dim=1)

        lcode = lcode + htooth

        feature = torch.cat([lcode, center, gcode[:, None].expand(-1, 28, -1)], dim=2)
        for i in range(3):
            feature = self.mlp[i](feature)
            if i < 2:
                feature = F.relu(feature)
                feature = F.dropout(feature, p=0.3, training=self.training)
        quat, pos = torch.split(feature, [4, 3], dim=2)
        quat = quat / quat.norm(dim=2, keepdim=True)
        return quat, pos
