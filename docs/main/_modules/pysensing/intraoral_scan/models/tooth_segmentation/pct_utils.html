import torch
from typing import Tuple



[docs]def square_distance(src: torch.Tensor, dst: torch.Tensor) -> torch.Tensor:
    r'''
    Calculate Euclidean distance between each pair of points from source and destination.

    The distance is calculated using the following formula:
    dist = ||src - dst||^2 = sum(src**2, dim=-1) + sum(dst**2, dim=-1) - 2 * (src^T * dst)

    Args:
        src (Tensor): Source points of shape [B, N, C], where B is batch size, N is number of points, C is coordinates (typically 3).
        dst (Tensor): Target points of shape [B, M, C], where B is batch size, M is number of target points, C is coordinates.

    Returns:
        Per-point squared distance of shape [B, N, M].
    '''
    B, N, _ = src.shape
    _, M, _ = dst.shape
    dist = -2 * torch.matmul(src, dst.permute(0, 2, 1))
    dist += torch.sum(src ** 2, -1).view(B, N, 1)
    dist += torch.sum(dst ** 2, -1).view(B, 1, M)
    return dist




[docs]def index_points(points: torch.Tensor, idx: torch.Tensor) -> torch.Tensor:
    r'''
    Index points using indices from a batch.

    Args:
        points (Tensor): Input points data of shape [B, N, C], where B is batch size, N is number of points, C is coordinates or features.
        idx (Tensor): Sample index data of shape [B, S], where S is the number of sampled points.

    Returns:
        Indexed points data of shape [B, S, C].
    '''
    device = points.device
    B = points.shape[0]

    view_shape = list(idx.shape)
    view_shape[1:] = [1] * (len(view_shape) - 1)    # [B, 1, 1, 1, 1, ...]

    repeat_shape = list(idx.shape)
    repeat_shape[0] = 1     # [1, S0, S1, ..., Sm-1]

    batch_indices = torch.arange(B, dtype=torch.long).to(device).view(view_shape).repeat(repeat_shape) # same shape with idx, to use advance indexing
    new_points = points[batch_indices, idx, :]
    return new_points # advance indexing will create a copy instead of a view




[docs]def knn_point(nsample: int, xyz: torch.Tensor, new_xyz: torch.Tensor) -> torch.Tensor:
    r'''
    Find the K nearest neighbors of each point in a given set.

    Args:
        nsample (int): Maximum number of neighbors to sample.
        xyz (Tensor): All points of shape [B, N, 3], where B is batch size, N is the number of points, and 3 is coordinates.
        new_xyz (Tensor): Query points of shape [B, N, 3]. Typically the same as `xyz` in segmentation tasks.

    Returns:
        Grouped points indices of shape [B, N, nsample].
    '''
    sqrdists = square_distance(new_xyz, xyz)    # [B, N, N]
    _, group_idx = torch.topk(sqrdists, nsample, dim = -1, largest=False, sorted=False)
    return group_idx # [B, N, k]




[docs]def sample_and_group(npoint: int, radius: float, nsample: int, xyz: torch.Tensor, points: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
    r'''
    Sample points and group their neighbors for feature extraction.

    Args:
        npoint (int): Number of output points (in segmentation, typically equal to N).
        radius (float): Radius of local region (not used in this implementation).
        nsample (int): Number of neighbors to sample for each point.
        xyz (Tensor): Input points position data of shape [B, N, 3].
        points (Tensor): Input points feature data of shape [B, N, D].

    Returns:
        Tuple (Tensor, Tensor):
            - new_xyz (Tensor): Sampled points position data of shape [B, N, k, 3].
            - new_points (Tensor): Sampled points data with concatenated normalized features of shape [B, N, k, 2 * D].
    '''
    B, N, C = xyz.shape
    
    xyz = xyz.contiguous()

    #######################################
    # For segmentation, do no change the number or order of the points
    assert(npoint == N)
    new_xyz = xyz
    new_points = points
    #######################################

    idx = knn_point(nsample, xyz, new_xyz) # [B, N, k]
    grouped_xyz = index_points(xyz, idx) # [B, N, k, 3]
    grouped_xyz_norm = grouped_xyz - new_xyz.view(B, N, 1, C)
    grouped_points = index_points(points, idx) # [B, N, k, D]
    grouped_points_norm = grouped_points - new_points.view(B, N, 1, -1)
    new_points = torch.cat([grouped_points_norm, new_points.view(B, N, 1, -1).repeat(1, 1, nsample, 1)], dim=-1) # [B, N, k, 2D]
    return new_xyz, new_points
