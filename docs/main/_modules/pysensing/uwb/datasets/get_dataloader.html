[docs]def load_localization_dataset(root, datasetname: str) -> Tuple[DataLoader, DataLoader]:
    r'''
    Load localization dataset for regression task.

    Args:
        dataset (:obj: 'string'): The name of the dataset that needs to be loaded and the loading mod. The choices are:
        {
        pedestrian_tracking_mod1_CIR
        pedestrian_tracking_mod2_CIR
        pedestrian_tracking_mod3_CIR
        pedestrian_tracking_mod1_Var
        pedestrian_tracking_mod2_Var
        pedestrian_tracking_mod3_Var
        }.
    
    Return:
        dataloaders for train, val and test datasets.
    Example:
        >>>train_loader, val_loader, test_loader = load_localization_dataset('pedestrian_tracking_mod1_CIR')
    '''
    
    if datasetname in ['pedestrian_tracking_mod1_CIR', 'pedestrian_tracking_mod2_CIR', 'pedestrian_tracking_mod3_CIR']:
        dataset_root =os.path.join(os.path.join(root, 'Pedestrian_Tracking'),'processed_data')
        if datasetname == 'pedestrian_tracking_mod1_CIR':
            print('Loading pedestrian tracking CIR dataset in mod 1...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod1', 'CIR')
        elif datasetname == 'pedestrian_tracking_mod2_CIR':
            print('Loading pedestrian tracking CIR dataset in mod 2...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod2', 'CIR')
        elif datasetname == 'pedestrian_tracking_mod3_CIR':
            print('Loading pedestrian tracking CIR dataset in mod 3...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod3', 'CIR')
        else:
            raise ValueError('The dataset does not exist.')
    elif datasetname in ['pedestrian_tracking_mod1_Var', 'pedestrian_tracking_mod2_Var', 'pedestrian_tracking_mod3_Var']:
        dataset_root =os.path.join(os.path.join(root, 'Pedestrian_Tracking'),'processed_data')
        if datasetname == 'pedestrian_tracking_mod1_Var':
            print('Loading pedestrian tracking Var dataset in mod 1...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod1', 'Var')
        elif datasetname == 'pedestrian_tracking_mod2_Var':
            print('Loading pedestrian tracking Var dataset in mod 2...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod2', 'Var')
        elif datasetname == 'pedestrian_tracking_mod3_Var':
            print('Loading pedestrian tracking Var dataset in mod 3...')
            X_train, Y_train, X_val, Y_val, X_test, Y_test = load_pedestrian_tracking(dataset_root, 'mod3', 'Var')
        else:
            raise ValueError('The dataset does not exist.')
    else:
        raise ValueError('The dataset should be in [pedestrian_tracking].')
        
    train_data = torch.tensor(X_train, dtype=torch.float32).permute(3,2,0,1) 
    # permute from (500, 2, 1, 91369) to (91369, 1, 500, 2)
    train_label = torch.tensor(Y_train, dtype=torch.long)
    val_data = torch.tensor(X_val, dtype=torch.float32).permute(3,2,0,1)
    # permute from (500, 2, 1, 16124) to (16124, 1, 500, 2)
    val_label = torch.tensor(Y_val, dtype=torch.long)
    test_data = torch.tensor(X_test, dtype=torch.float32).permute(3,2,0,1)
    # permute from (500, 2, 1, 53635) to (53635, 1, 500, 2)
    test_label = torch.tensor(Y_test, dtype=torch.long)
    train_set = torch.utils.data.TensorDataset(train_data, train_label)
    val_set = torch.utils.data.TensorDataset(val_data, val_label)
    test_set = torch.utils.data.TensorDataset(test_data, test_label)
    train_loader = torch.utils.data.DataLoader(train_set, batch_size=64, shuffle=True)
    val_loader = torch.utils.data.DataLoader(val_set, batch_size=64, shuffle=False)
    test_loader = torch.utils.data.DataLoader(test_set, batch_size=256, shuffle=False)
    
    return train_loader, val_loader, test_loader



[docs]def load_occupancy_detection_dataset(root, datasetname: str) -> Tuple[DataLoader, DataLoader]:
    r'''
    Load occupancy detection dataset for binary classification task.

    Args:
        dataset (:obj: 'string'): The name of the dataset that needs to be loaded. The choices are:
        {
        'nlos_human_detection_raw_dynamic'
        'nlos_human_detection_raw_static'
        }.

    Return:
        dataloaders for train and val datasets.

    Example:
        >>>train_loader, test_loader = load_occupancy_detection_dataset('nlos_human_detection_raw_dynamic')
    '''
    if datasetname.split('_')[0] == 'nlos':
        dataset_root = os.path.join(root, 'nlos_human_detection')
        dataset = nlos_human_detection(dataset_root)
        if datasetname == 'nlos_human_detection_raw_dynamic':
            print('Loading nlos_human_detection time domain dynamic dataset...')
            data, label = dataset.load_raw_data('dynamic')

        elif datasetname == 'nlos_human_detection_raw_static':
            print('Loading nlos_human_detection time domain static dataset...')
            dataset_root = os.path.join(root, 'nlos_human_detection')
            data, label = dataset.load_raw_data('static')
        
        else:
            raise ValueError('The dataset does not exist.')

    else:
        raise ValueError('The dataset should be in [nlos_human_detection].')
    
    train_data, val_data, train_label, val_label = train_test_split(data, label, test_size=0.2, stratify=label)
    train_data = torch.tensor(train_data, dtype=torch.float32)
    train_label = torch.tensor(train_label, dtype=torch.long)
    val_data = torch.tensor(val_data, dtype=torch.float32)
    val_label = torch.tensor(val_label, dtype=torch.long)
    train_set = torch.utils.data.TensorDataset(train_data, train_label)
    test_set = torch.utils.data.TensorDataset(val_data, val_label)
    train_loader = torch.utils.data.DataLoader(train_set, batch_size=64, shuffle=True)
    test_loader = torch.utils.data.DataLoader(test_set, batch_size=256, shuffle=False)

    return train_loader, test_loader



[docs]def load_har_dataset(root, datasetname: str) -> Tuple[DataLoader, DataLoader]:
    r'''
    Load human activity recognition dataset for multi-class classification task.

    Args:
        dataset (:obj: 'string'): The name of the dataset that needs to be loaded. 
        The choices are:{
        'Sleepposenet_dataset1'
        'Sleepposenet_dataset2_session1_ceiling'
        'Sleepposenet_dataset2_session1_wall'
        'Sleepposenet_dataset2_session1_all'
        'Sleepposenet_dataset2_session2_ceiling'
        'Sleepposenet_dataset2_session2_wall'
        'Sleepposenet_dataset2_session2_all'
        'Sleepposenet_dataset2_sessionALL_ceiling'
        'Sleepposenet_dataset2_sessionALL_wall'
        'Sleepposenet_dataset2_sessionALL_all'
        }.

    Return:
        dataloaders for train and val datasets.

    Example:
        >>>train_loader, test_loader = load_har_dataset('Sleepposenet_dataset1')
    '''
    if datasetname.split('_')[0] == 'Sleepposenet':

        dataset_root = os.path.join(root, 'sleep_pose_net_data')
        dataset = SleepposenetDataset(dataset_root, preprocess=True)

        if datasetname == 'Sleepposenet_dataset1':
            print('Loading Sleep Pose Net Dataset 1 ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_1()

        elif datasetname == 'Sleepposenet_dataset2_session1_ceiling':
            print('Loading Sleep Pose Net Dataset 2 session 1 ceiling ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(1, 'ceiling')

        elif datasetname == 'Sleepposenet_dataset2_session1_wall':
            print('Loading Sleep Pose Net Dataset 2 session 1 wall ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(1, 'wall')

        elif datasetname == 'Sleepposenet_dataset2_session1_all':
            print('Loading Sleep Pose Net Dataset 2 session 1 all ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(1, 'all')

        elif datasetname == 'Sleepposenet_dataset2_session2_ceiling':
            print('Loading Sleep Pose Net Dataset 2 session 2 ceiling ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(2, 'ceiling')

        elif datasetname == 'Sleepposenet_dataset2_session2_wall':
            print('Loading Sleep Pose Net Dataset 2 session 2 wall ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(2, 'wall')

        elif datasetname == 'Sleepposenet_dataset2_session2_all':
            print('Loading Sleep Pose Net Dataset 2 session 2 all ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(2, 'all')

        elif datasetname == 'Sleepposenet_dataset2_sessionALL_ceiling':
            print('Loading Sleep Pose Net Dataset 2 session ALL ceiling ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(3, 'ceiling')

        elif datasetname == 'Sleepposenet_dataset2_sessionALL_wall':
            print('Loading Sleep Pose Net Dataset 2 session ALL wall ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(3, 'wall')

        elif datasetname == 'Sleepposenet_dataset2_sessionALL_all':
            print('Loading Sleep Pose Net Dataset 2 session ALL all ...')
            X_diff_train, X_diff_test, X_wrtft_train, X_wrtft_test, y_train, y_test = dataset.load_dataset_2(3, 'all')

        else:
            raise ValueError('The dataset does not exist.')
        
    else:
        raise ValueError('The dataset should be in [Sleepposenet].')
    
    train_td_data = torch.tensor(X_diff_train, dtype=torch.float32)
    train_wrtft_data = torch.tensor(X_wrtft_train, dtype=torch.float32)
    train_label = torch.tensor(y_train, dtype=torch.long)
    val_td_data = torch.tensor(X_diff_test, dtype=torch.float32)
    val_wrtft_data = torch.tensor(X_wrtft_test, dtype=torch.float32)
    val_label = torch.tensor(y_test, dtype=torch.long)
    train_set = torch.utils.data.TensorDataset(train_td_data, train_wrtft_data, train_label)
    test_set = torch.utils.data.TensorDataset(val_td_data, val_wrtft_data, val_label)
    train_loader = torch.utils.data.DataLoader(train_set, batch_size=32, shuffle=True)
    test_loader = torch.utils.data.DataLoader(test_set, batch_size=64, shuffle=False)
    
    return train_loader, test_loader
