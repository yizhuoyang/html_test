import numpy as np
from skimage import filters
from scipy import signal
from tqdm import tqdm
from sklearn.preprocessing import StandardScaler


[docs]def time_difference_transform(x: np.ndarray, norm: bool) -> np.ndarray:
    r'''
    This method is based on an assumption that every obstacles in the background \
    are static or not changing its shape over the time. Therefore, \
    the differentiation along the slow time dimension can extract only \
    the information of a moving human.

    Args:
        x (:obj: 'numpy.array'): The input data with shape (num_samples, fast_time_indices, slow_time_indices).

    Return:
        x_diff: The (normalized) data after time difference selection.

    Example:
        >>>x_diff = time_difference_selection(x)
    '''
    space, time = x.shape[1], x.shape[2]
    x_diff = np.diff(x, n=1,axis=2)
    x_diff = x_diff.reshape(x_diff.shape[0], space*(time-1))
    if norm:
        scaler_diff = StandardScaler()
        scaler_diff.fit(x_diff)
        x_diff = scaler_diff.transform(x_diff)
        x_diff = x_diff.reshape(x_diff.shape[0], 1, space, time-1).astype('float32')
    else:
        x_diff = x_diff.reshape(x_diff.shape[0], 1, space, time-1).astype('float32')

    return x_diff




[docs]def wrtft(x: np.ndarray, NFFT: int, stride: int) -> np.ndarray:
    r'''
    Perform weighted RTF transform on the 2D input data.

    Args:
        x (:obj: 'numpy.array'): The input data with shape (fast_time_indices, slow_time_indices).
        NFFT (:obj: 'int''): A positive integer, it is the number of data points used in each block for the FFT.
        stride (:obj: 'int'): A positive integer, it is the number of overlapping points between segments.

    Return:
        F: The data after weighted RTF transform.

    Example:
        >>>F = wrtft(x, NFFT, stride)
    '''
    R = x # select frame range 
    val = filters.threshold_otsu(R)
    B = np.copy(R)

    for i in range(R.shape[0]):
        for j in range(R.shape[1]):
            if R[i][j] < val:
                B[i][j] = 0
            else :
                B[i][j] = 1

    min_r = 200
    max_r = -1
    for i in range(B.shape[0]):
        for j in range(B.shape[1]):
            if B[i][j]:
                min_r = min(min_r,i)
                max_r = max(max_r,i)
    Em = []
    for i in range(R.shape[0]):
        Em.append(np.sum(R[i,:]**2))
    Em = np.array(Em)            
    sigma = Em[min_r:max_r]/sum(Em[min_r:max_r]) 

    F = np.zeros((NFFT//2+1,(R.shape[1]-NFFT)//stride+1))
    for m in range(min_r,max_r):
        _,_,arr = signal.stft(R[m,:], fs = 1, nperseg=NFFT, noverlap = NFFT - stride, padded = False, boundary = None)
        arr = np.abs(arr)
        F += sigma[m-min_r]*arr
        del arr
    return F



[docs]def apply_wrtft(x: np.ndarray, NFFT: int, stride: int) -> np.ndarray:
    r'''
    Apply weighted RTF transform on the 3D input data (1st dim is n_samples).
    
    Args:
        x (:obj: 'numpy.array'): The input data with shape (num_samples, fast_time_indices, slow_time_indices).
        NFFT (:obj: 'int'): Same as the NFFT in the wrtft function.
        stride (:obj: 'int'): Same as the stride in the wrtft function.
    
    Return:
        res: The data after weighted RTF transform.
    
    Example:
        >>>res = apply_wrtft(x, NFFT, stride)
    '''
    res = []
    cnt = 0
    for dat in tqdm(x,position = 0):
        F = wrtft(dat, NFFT, stride)
        res.append(F)
        cnt += 1
    return np.array(res)



[docs]def  weighted_rtf_transform(x: np.ndarray, NFFT: int, stride: int, norm: bool) -> np.ndarray:
    r'''
    The weighted RTF transform is a method to extract the most important features \
    from the input data. The weights are used to multiply with the input data in \
    the slow time dimension. The output data is the summation of the multiplication \
    of the input data and the weights.

    Args:
        x (:obj: 'numpy.array'): The input data with shape (num_samples, fast_time_indices, slow_time_indices).

    Return:
        x_wrtft: The (normalized) data after weighted RTF transform.

    Example:
        >>>x_wrtft = weighted_rtf_transform(x, NFFT, stride, norm)
    '''
    x_wrtft = apply_wrtft(x, NFFT, stride)
    range_bins, time_bins = x_wrtft.shape[1], x_wrtft.shape[2]
    x_wrtft = x_wrtft.reshape(-1, range_bins*time_bins)
    if norm:
        scaler_wrtft = StandardScaler()
        scaler_wrtft.fit(x_wrtft)
        x_wrtft = scaler_wrtft.transform(x_wrtft)
        x_wrtft = x_wrtft.reshape(-1, 1, range_bins, time_bins)
    else:
        x_wrtft = x_wrtft.reshape(-1, 1, range_bins, time_bins)
    
    return x_wrtft
