[docs]def test(model, tensor_loader, criterion, device):
    r'''
    Test the model on the validation set.

    Args:
        model (:obj: 'torch.nn.Module'): the model to be tested.
        tensor_loader (:obj: 'torch.utils.data.DataLoader'): the data loader of the validation set.
        criterion (:obj: 'torch.nn.Loss'): the criterion to calculate the loss.
        device (:obj: 'torch.device'): the device to run the model
    
    Returns:
        None
    '''
    model.eval()
    test_acc = 0
    test_loss = 0
    for data in tensor_loader:
        x, labels = data
        x = x.to(device)
        labels = labels.to(device)
        
        labels = labels.type(torch.LongTensor)

        outputs = model(x)
        outputs = outputs.type(torch.FloatTensor)
        outputs.to(device)
        
        loss = criterion(outputs,labels)
        predict_y = torch.argmax(outputs,dim=1).to(device)
        # print(predict_y)
        accuracy = (predict_y == labels.to(device)).sum().item() / labels.size(0)
        test_acc += accuracy
        test_loss += loss.item() * x.size(0)
    test_acc = test_acc/len(tensor_loader)
    test_loss = test_loss/len(tensor_loader.dataset)
    print("validation accuracy:{:.4f}, loss:{:.5f}".format(float(test_acc),float(test_loss)))
    return



[docs]def train(model, train_loader, val_loader, num_epochs, learning_rate, criterion, device):
    r'''
    Train the model and evaluate the model every 10 epoches.

    Args:
        model (:obj: 'torch.nn.Module'): the model to be trained.
        train_loader (:obj: 'torch.utils.data.DataLoader'): the data loader of the training set.
        val_loader (:obj: 'torch.utils.data.DataLoader'): the data loader of the validation set.
        num_epochs (:obj: 'int'): the number of epoches to train the model.
        learning_rate (:obj: 'float'): the learning rate of the optimizer.
        criterion (:obj: 'torch.nn.Loss'): the criterion to calculate the loss.
        device (:obj: 'torch.device'): the device to run the model
    
    Returns:
        None
    '''
    model = model.to(device)
    optimizer = torch.optim.Adam(model.parameters(), lr = learning_rate)
    lr_scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=500, gamma=0.1)
    for epoch in range(num_epochs):
        model.train()
        epoch_loss = 0
        epoch_accuracy = 0
        for data in train_loader:
            x, labels = data
            x = x.to(device)
            labels = labels.to(device)
            
            labels = labels.type(torch.LongTensor)
            
            optimizer.zero_grad()
            outputs = model(x)
            outputs = outputs.to(device)
            outputs = outputs.type(torch.FloatTensor)
            loss = criterion(outputs,labels)
            loss.backward()
            optimizer.step()
            
            epoch_loss += loss.item() * x.size(0)
            predict_y = torch.argmax(outputs,dim=1).to(device)
            
            epoch_accuracy += (predict_y == labels.to(device)).sum().item() / labels.size(0)
            # break
        epoch_loss = epoch_loss/len(train_loader.dataset)
        epoch_accuracy = epoch_accuracy/len(train_loader)
        print('Epoch:{}, Accuracy:{:.4f},Loss:{:.9f}'.format(epoch+1, float(epoch_accuracy),float(epoch_loss)))
        # break
        if (epoch+1) % 10 == 0:
            test(
                model=model,
                tensor_loader=val_loader,
                criterion=criterion,
                device= device
            )
            # print(a)
        lr_scheduler.step()

    return



[docs]def nlos_human_detection_training(root, dataset_name: str, model_name: str, num_epochs: int, learning_rate: float, save_weights: bool):
    r'''
    Trigger the train process of the human detection model.

    Args:
        dataset_name (:obj: 'string'): the name of the dataset
        model_name (:obj: 'string'): the name of the model
        num_epochs (:obj: 'int'): the number of epoches to train the model.
        learning_rate (:obj: 'float'): the learning rate of the optimizer.
        save_weights (:obj: 'bool'): whether to save the weights of the model.
    
    Returns:
        None
    '''
    train_loader, test_loader = load_occupancy_detection_dataset(root, dataset_name)
    criterion = nn.CrossEntropyLoss()
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model = load_occupancy_detection_model('nlos_human_detection', model_name)
    model.to(device)
    train(
        model=model,
        train_loader= train_loader,
        val_loader= test_loader,
        num_epochs= num_epochs,
        learning_rate=learning_rate,
        criterion=criterion,
        device=device
            )
    if save_weights:
        torch.save(model.state_dict(), 'nlos_human_detection_weights.pth')
    return
