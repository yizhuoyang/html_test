[docs]def load_har_model(dataset_name: str, model_name: str):
    r'''
    Load the model for HAR task.

    Args:
        dataset_name (:obj: 'string'): 'sleep_pose_net_dataset1' or 'sleep_pose_net_dataset2'
        model_name (:obj: 'string'): 'sleepposenet'
    
    Return:
        model (:obj: 'torch.nn.Module'): the model for HAR
    '''
    classes = {'sleep_pose_net_dataset1': 6, 'sleep_pose_net_dataset2': 7}
    if model_name == 'sleepposenet':
        model = sleepposenet(feature_dim=128, num_classes=classes[dataset_name])
    else:
        raise ValueError('Invalid model name.')
    return model



[docs]def load_occupancy_detection_model(dataset_name: str, model_name: str):
    r'''
    Load the model for occupancy detection task.

    Args:
        dataset_name (:obj: 'string'): 'nlos_human_detection'
        model_name (:obj: 'string'): 'autoencoder' or 'lstm'
    
    Return:
        model (:obj: 'torch.nn.Module'): the model for occupancy
    '''
    if dataset_name == 'nlos_human_detection':
        if model_name == 'autoencoder':
            model = occupancy_detection_autoencoder(hidden_size= 16)
        elif model_name == 'lstm':
            model = occupancy_detection_lstm()
        else:
            raise ValueError('Invalid model name.')
    else:
        raise ValueError('Invalid dataset name.')
    return model

def load_pretrain_weights(model, dataset_name, model_name, device):
    r'''
    Load pretrain models from files.

    Args:
        model (:obj: 'nn.Module'): the model for the task.
        dataset_name (:obj: 'str'): the dataset name.
        model_name (:obj: 'str'): the model name.
        device: str, the device to load the model. Default: 'cpu'.

    Returns:
        model: the pretrain model for the task.
    '''
    # pretrain_root = '../model/pretrain/'
    pretrain_root = 'https://pysensing.oss-ap-southeast-1.aliyuncs.com/pretrain/uwb'
    if dataset_name in ['sleep_pose_net_dataset1', 'sleep_pose_net_dataset2']:
        folder = pretrain_root + '/HAR'
        pretrain_path = folder + '/' +  dataset_name + '.pth'
        model.load_state_dict(torch.hub.load_state_dict_from_url(pretrain_path,progress=True,map_location=device))
    elif dataset_name in ['nlos_human_detection_static', 'nlos_human_detection_dynamic']:
        folder = pretrain_root + '/occupancy_detection'
        if model_name == 'autoencoder':
            if dataset_name == 'nlos_human_detection_static':
                pretrain_path = folder + '/' + 'nlos_detection_static_autoencoder.pth'
            elif dataset_name == 'nlos_human_detection_dynamic':
                pretrain_path =  folder + '/' + 'nlos_detection_dynamic_autoencoder.pth'
            else:
                raise ValueError('Dataset name for nlos_human_detection is invalid. Please choose from [nlos_human_detection_static,nlos_human_detection_dynamic].')
        elif model_name == 'lstm':
            if dataset_name == 'nlos_human_detection_static':
                raise ValueError('lstm pretrained model for nlos_human_detection_static is not provided yet.')
            elif dataset_name == 'nlos_human_detection_dynamic':
                pretrain_path =  folder + '/' + 'nlos_detection_dynamic_lstm.pth'
        else:
            raise ValueError('Model name for nlos_human_detection is invalid. Please choose from [autoencoder,lstm].')
        model.load_state_dict(torch.hub.load_state_dict_from_url(pretrain_path,progress=True,map_location=device))
    elif dataset_name == 'human_tracking':
        folder = pretrain_root + '/' + 'localization'
        if model_name == 'Var_model':
            pretrain_path =  folder + '/' + 'Pedestrian_Tracking_Var_mod1_CNN.pth'
        elif model_name == 'CIR_model':
            pretrain_path =  folder + '/' + 'Pedestrian_Tracking_CIR_mod1_CNN.pth'
        else:
            raise ValueError('Model name for human_tracking is invalid. Please choose from [Var_model,CIR_model].')
        model.load_state_dict(torch.hub.load_state_dict_from_url(pretrain_path,progress=True,map_location=device))
    else:
        raise ValueError('Dataset name is invalid. Please choose from [sleep_pose_net_dataset1,sleep_pose_net_dataset2,nlos_human_detection,human_tracking].')
    return model
