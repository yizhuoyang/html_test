[docs]class MGesture(Dataset):
    r"""
    Implementation of "M-Gesture: Person-Independent Real-Time In-Air Gesture Recognition Using Commodity Millimeter Wave Radar".

    Point cloud gesture dataset collected using FMCW mmWave Radar, TI-IWR1443 single-chip 76-GHz to 81-GHz mmWave sensor evaluation module. 2 scenarios are included: 
    short range (i.e. Human-Radar Distance(HRD) < 0.5 m) and long range (i.e. 2m < HRD < 5m); Only long-range gesture recognition 
    is supported as only long-range dataset contain point cloud data.

        Args:
        root (str of Path): Path to the .csv file

        split (str): Split of the dataset. Selected from ["train", "val", "test"] 

        class_num (int): The number of actions selected for training and testing, 

    Reference: 
        https://github.com/fengxudi/mmWave-gesture-dataset
    """
    def __init__(self,
                root='/home/junqiao/projects/data/mGesture',
                split='Test',
                class_num = 4):
        self.root = os.path.join(root, "gesture_dataset/long_range_gesture/long_SEP/") 
        self.split = split
        self.classes = ['knock', 'rotate', 'rswipe', 'lswipe']
        self.class_num = class_num

        self.npoints = 50
        self.frames_together = 28
        self.sliding = 1

        self.pc_all = []
        self.labels_all = []
        self.slicing_marker = []

        # long-range dataset
        path = os.path.join(self.root, "*")
        for f1 in glob.glob(path):
            subject = f1.split("/")[-1]
            if split == "Train" and subject == "008":
                continue
            if split == "Test" and subject in ["001","002","003","004","005","006","007"]:
                continue
            for f2 in glob.glob(os.path.join(f1, "*.csv")):

                # load and process csv file
                pc_data = get_data_csv(f2)
                sequence_length = len(pc_data)
                subject = int(f2.split("/")[-1][:-4].split("_")[-2])
                action = f2.split("/")[-1][:-4].split("_")[-1]
                label = self.classes.index(action)
                self.slicing_marker.append(sequence_length)

                for frame in pc_data:
                    frame_data = np.array(pc_data[frame])

                    # preprocessing
                    frame_data = padding(frame_data, npoint=self.npoints)

                    self.pc_all.append(frame_data)
                    self.labels_all.append(label)
        
        self.idxs = sliding_window(index_type="idx", frames_together = self.frames_together, sliding = self.sliding, slicing_markers=self.slicing_marker)
        print(f"Generating {len(self.idxs)} samples...")




    def __len__(self):
        return len(self.idxs)

    def __getitem__(self, idx):
        r'''
        Returns:
            (input, label)[np.ndarray, int]:
            "input": The voxel images from preprocessed radar point clouds. (np.ndarray) the shape of voxel image: [frame_together, sliding, width, height]. Default [60, 10, 32, 32]
            "label": The class label. (int) the index of actions indicating class label.
                
        Example:
            >>> har_train_dataset = radHAR(dataset_root, split='Train')
            >>> index = 9
            >>> sample= har_train_dataset.__getitem__(index)
        '''

        start = self.idxs[idx]
        end = start + self.frames_together

        inputs = self.pc_all[start:end]
        label = self.labels_all[start]

        inputs = np.stack(inputs, axis=0)
        

        return inputs, label

                
                
                
                
    


def get_data_csv(path):
    '''
    input: list of paths
    output: np array for mmWave PC: dict of list
    '''
    data = dict()

    # read all csv into lists
    frame_all = []
    x_all = []
    y_all = []
    z_all = []
    doppler_all = []
    intensity_all = []

    with open(path) as file_obj: 
        reader_obj = csv.reader(file_obj) 
        # skip header
        next(reader_obj)
        for row in reader_obj: 
            frame_all.append(int(row[0]))
            pc_num = int(row[1])
            x_all.append(float(row[2]))
            y_all.append(float(row[3]))
            z_all.append(float(row[4]))
            doppler_all.append(float(row[5]))
            intensity_all.append(float(row[6]))

    for i in range(len(frame_all)):
        if int(frame_all[i]) in data:
            data[frame_all[i]].append([x_all[i],y_all[i],z_all[i],doppler_all[i],intensity_all[i]])
        else:
            data[frame_all[i]]=[]
            data[frame_all[i]].append([x_all[i],y_all[i],z_all[i],doppler_all[i],intensity_all[i]])

    return data



[docs]def load_hgr_dataset(dataset, dataset_root=None):
    r"""
    This function provide quick construct train-test dataset based on dataset name.

    Args:
        dataset (str): Name of dataset.

        dataset_root (str of path): Root dir of the data set.

    Return:
        train_dataset (torch.utils.data.Dataset): train split of dataset using pytorch.

        test_dataset (torch.utils.data.Dataset): test split of dataset using pytorch.

    """
    if dataset == 'M-Gesture':
        dataset_root = check_dataset(dataset, dataset_root)

        print('using dataset: M-Gesture DATA')
        train_dataset = MGesture(root=dataset_root, split='Train')
        test_dataset = MGesture(root=dataset_root, split='Test')
        

    else:
        raise ValueError("Unsupported dataset. Please choose from 'M-Gesture'.")

    return train_dataset, test_dataset
