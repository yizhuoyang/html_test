from tqdm import tqdm

from pysensing.mmwave.PC.preprocessing.uniform import load_from_raw
from pysensing.mmwave.PC.preprocessing.sliding_window import sliding_window
from pysensing.mmwave.PC.dataset.utils import check_dataset


[docs]class radHAR(Dataset):
    r"""
    Implementation of "RadHAR: Human Activity Recognition from Point Clouds Generated through a Millimeter-wave Radar".

    Point cloud dataset collected using mmWave Radar. The data is collected using the ROS package for TI mmWave radar developed by radar-lab.
    The data is partitioned into the folders train and test. Each train and test folder further contains the folders which have respective activity classes.

    Args:
        root (str of Path): Path to the .txt file

        split (str): Split of the dataset. Selected from ["train", "val", "test"] 

        actions (python list of str): The selected actions for training and testing, the index indicates the class labels. For example, ['boxing','jack','jump','squats','walk'] means 'boxing' = 0, 'jack' = 1, ..., 'walk' = 4.

        load_from_preprocess (bool): The indicator for loading options: if True, load from proprocessed .npz files; if False, load from raw txt files.


    Reference: 
        https://github.com/nesl/RadHAR
    """
    def __init__(self,
                 root='/home/junqiao/projects/humanHAR/RadHAR/Data/',
                 split='Train',
                 actions =['boxing','jack','jump','squats','walk'],
                 load_from_preprocess = False):
        self.root = root
        self.split = split
        self.actions = actions

        self.frame_together = 60
        self.sliding = 10






        if load_from_preprocess == False:
            if len(glob.glob(os.path.join(self.root, self.split, "data.npz"))) != 0:
                data = np.load(os.path.join(self.root, self.split, "data.npz"))
                self.voxels_all, self.slicing_marker, self.labels_all = data['arr_0'], data['arr_1'], data['arr_2']
                self.idxs = sliding_window(index_type="idx", frames_together = self.frame_together, sliding = self.sliding, slicing_markers=self.slicing_marker)


            else:
                parent_dir = os.path.join(root, split)
                self.voxels_all, self.slicing_marker, self.labels_all = load_from_raw(parent_dir,sub_dirs=self.actions)
                self.idxs = sliding_window(index_type="idx", frames_together = self.frame_together, sliding = self.sliding, slicing_markers=self.slicing_marker)

                
                

           




        if load_from_preprocess == True:
            # initialize dataframe
            self.prefix ="Train_Data_voxels_"
            self.actions = ['boxing','jack','jump','squats','walk']
            self.mmPC = np.empty((0, 60, 10, 32, 32) )
            self.label = np.empty((0, ) )


            for action in self.actions:
                print(os.path.join(self.root, self.split, self.prefix + action + ".npz"))
                self.mmPC = np.vstack([self.mmPC, np.load(os.path.join(self.root, self.split, self.prefix + action + ".npz"))['arr_0']])
                self.label = np.append(self.label, np.load(os.path.join(self.root, self.split, self.prefix + action + ".npz"))['arr_1'])
                print(self.mmPC.shape, self.label.shape)

    def __len__(self):
        return self.idxs.shape[0]

    def __getitem__(self, idx):
        r'''
        Returns:
            (input, label)[np.ndarray, int]: 
            "input": The voxel images from preprocessed radar point clouds. (np.ndarray) the shape of voxel image: [frame_together, sliding, width, height]. Default [60, 10, 32, 32]
            "label": The class label. (int) the index of actions indicating class label.
                    
        Example:
            >>> har_train_dataset = radHAR(dataset_root, split='Train')
            >>> index = 9
            >>> sample= har_train_dataset.__getitem__(index)
        '''
        start = self.idxs[idx]
        end = start + self.frame_together

        inputs = self.voxels_all[start:end]
        label = self.labels_all[start]
        label = self.actions.index(label)




        return inputs, label

    

[docs]def load_har_dataset(dataset, dataset_root=None):
    r"""
    This function provide quick construct train-test dataset based on dataset name.

    Args:
        dataset (str): Name of dataset.

        dataset_root (str of path): Root dir of the data set.

    Return:
        train_dataset (torch.utils.data.Dataset): train split of dataset using pytorch.

        test_dataset (torch.utils.data.Dataset): test split of dataset using pytorch.

    """
    if dataset == 'radHAR':
        dataset_root = check_dataset(dataset, dataset_root)
        print('using dataset: radHAR DATA')
        train_dataset = radHAR(dataset_root, split='Train')
        test_dataset = radHAR(dataset_root, split='Test')

    else:
        raise ValueError("Unsupported dataset. Please choose from 'radHAR'.")

    return train_dataset, test_dataset
