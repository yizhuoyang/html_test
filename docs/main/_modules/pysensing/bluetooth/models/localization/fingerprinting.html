[docs]class MLP(nn.Module):
    r'''
    The typical MLP model with three :obj:`Linear` layers and two :obj:`LeakyReLU` layers.
    Default negative slope of the :obj:`LeakyReLU` layers is set to 0.2.

    Args:
        dim_in (:obj:`Int`): Dimension of the input BLE vector.
        dim_hidden (:obj:`Int`): Dimension of the hidden feature.
        dim_out(:obj:`Int`): Dimension of the output, which is usually ``2`` or ``3``.
    '''
    def __init__(self, dim_in, dim_out=2, dim_hidden=[16, 64]):
        super(MLP, self).__init__()
        self.dim_input = dim_in
        self.dim_output = dim_out

        self.network = nn.ModuleList()

        _dim_in = dim_in
        for layer_idx in range(len(dim_hidden)):
            _dim_out = dim_hidden[layer_idx]
            self.network.append(nn.Linear(_dim_in, _dim_out))
            self.network.append(nn.LeakyReLU(0.2))
            self.network.append(nn.Linear(_dim_out, _dim_out))
            self.network.append(nn.LeakyReLU(0.1))
            _dim_in = _dim_out
        self.network.append(nn.Linear(_dim_in, dim_out))


[docs]    def forward(self, x):
        for _layer in self.network:
            x = _layer(x)
        return x

        for idx, _layer in enumerate(self.network):
            if idx + 1 < len(self.network):
                x = _layer(x)
        return x




[docs]class CNN(nn.Module):
    r'''
    The 1D scanning CNN model includes feature encoder, convolution layer and the projection layer.
    The feature encoder consists of two :obj:`Linear` layers with :obj:`LeakyReLU` and :obj:`Tanh` activation functions.
    The convolution layer consists of several stacked combination of :obj:`Conv1D` and :obj:`LeakyReLU` operations.
    The projection layer consists of two :obj:`Linear` operations.

    Args:
        dim_in (:obj:`Int`): Dimension of the input BLE vector.
        dim_hidden (:obj:`Int`): Dimension of the hidden feature.
        dim_out(:obj:`Int`): Dimension of the output, which is usually ``2`` or ``3``.

    Note:
        - When using the final activation operation, :obj:`ReLU` is not recommended
          because the location may probably be non-positive values.
    '''
    def __init__(self, dim_in, dim_out=2, dim_embed=64, channels=[16, 32]):
        super(CNN, self).__init__()
        self.dim_in = dim_in
        self.dim_out = dim_out
        self.dim_embed = dim_embed
        self.channels = channels

        self.encoder = nn.Sequential(
            OrderedDict([
                ("linear1", nn.Linear(self.dim_in, self.dim_embed*2)),
                ("activation1", nn.LeakyReLU(0.2)),
                ("linear2", nn.Linear(self.dim_embed*2, self.dim_embed)),
                ("activation2", nn.Tanh())
            ])
        )
        self.dropout = nn.Dropout(p=0.2)
        self.CNN = nn.ModuleList()
        chn_in = 1
        for layer_idx in range(len(channels)):
            chn_out = channels[layer_idx]
            self.CNN.append(nn.Conv1d(chn_in, chn_out, kernel_size=5, stride=3))
            self.CNN.append(nn.LeakyReLU(0.2))
            chn_in = chn_out

        self.regressor = nn.Sequential(
            nn.LazyLinear(16),
            nn.LeakyReLU(0.2),
            nn.LazyLinear(dim_out),
        )


[docs]    def forward(self, x):
        x = self.encoder(x)
        # x = self.dropout(x)
        x = torch.unsqueeze(x, dim=1)
        for _layer in self.CNN:
            x = _layer(x)
        x = torch.flatten(x, 1)
        x = self.regressor(x)
        return x

[docs]class WKNN:
    r'''
    The Weighted K-Nearest Neighbor algorithm in which the Euclidean distance is set to be the similarity metric.

    Args:
        database_position (:obj:`Numpy.ndarray`): The positions of all training reference points.
            The shape has to be (N, L), where :math:`N` is the number of reference points, and :math:`L`
            is the dimension of the coordinate.
        database_ble (:obj:`Numpy.ndarray`): The BLE vectors of all training reference points.
            The shape has to be (N, R), where :math:`R` is the dimension of the whole BLE vector.

    Note:
        - This function only works in the inference mode and takes only one BLE vector as input at the same time.
        - First call of the class will create the object of WKNN, and second call of the object will preduce the result.

    '''
    def __init__(self, database_position, database_ble):
        super(WKNN, self).__init__()
        self.database_position = database_position
        self.database_ble = database_ble


[docs]    def compute_similarity(self, point_query, point_support):
        r'''
        Perform L2 Norm (Euclidean distance) similarity check between a query BLE and the stored BLE vectors.

        Args:
            point_query (:obj:`ndarray`): Query BLE vector.
            point_support (:obj:`ndarray`): Stored BLE vectors.

        Returns:
            (:obj:`ndarray`): Similarity.
        '''
        ble_err = point_query - point_support
        abs_err = np.linalg.norm(ble_err)

        abs_err += 1e-4 if abs_err == 0 else 0
        similarity = 1 / abs_err
        return similarity


    def __call__(self, input_ble, K=5):
        r'''
        Perform the WKNN algorithm.

        Args:
            input_ble (:obj:`ndarray`): Input BLE vector.
            K (:obj:`int`): K neareast neighbors will be used.

        Returns:
            (:obj:`ndarray`): Estimated position.
        '''
        self.num_best = K
        len_base = len(self.database_ble)
        similarity = np.zeros(len_base)
        for i in range(0, len_base):
            similarity[i] = self.compute_similarity(input_ble, self.database_ble[i])
        idx_similarity = np.argsort(similarity, axis=-1, kind='quicksort', order=None)[::-1]
        similarity_ordered = similarity[idx_similarity]
        neighbor_position = self.database_position[idx_similarity[:self.num_best], :]
        neighbor_similarity = similarity[idx_similarity[:self.num_best]]
        neighbor_weight = neighbor_similarity / sum(neighbor_similarity)
        estimate_position = np.average(neighbor_position, weights=neighbor_weight, axis=0)
        return estimate_position




[docs]class LSTM(nn.Module):
    r'''
    The LSTM localization model includes feature encoder, LSTM layer and the projection layer.
    The feature encoder consists of two :obj:`Linear` layers with :obj:`LeakyReLU` and :obj:`Tanh` activation functions.
    The LSTM layer is an instance of ``torch.nn.LSTM`` method.
    The projection layer is an instance of ``torch.nn.Linear`` method.

    Args:
        dim_in (:obj:`Int`): Dimension of the input BLE vector.
        dim_embed (:obj:`Int`): Dimension of the embedding feature.
        dim_lstm (:obj:`Int`): Dimension of the LSTM model hidden feature.
        dim_out (:obj:`Int`): Dimension of the output, which is usually ``2`` or ``3``.
        num_layers (:obj:`Int`): Number of LSTM layers. Default: 1.

    Note:
        - When using the final activation operation, :obj:`ReLU` is not recommended
          because the location may probably be non-positive values.
    '''
    def __init__(self, dim_in, dim_out=2, dim_embed=64, dim_lstm=32, num_layers=1):
        super(LSTM, self).__init__()
        self.dim_in = dim_in
        self.dim_embed = dim_embed
        self.dim_lstm = dim_lstm
        self.dim_out = dim_out

        self.embedding = nn.Sequential(
            OrderedDict([
                ("linear1", nn.Linear(dim_in, dim_embed)),
                ("activation1", nn.LeakyReLU(0.2)),
                ("linear2", nn.Linear(dim_embed, dim_embed)),
                ("activation2", nn.Tanh())
            ])
        )

        self.lstm = nn.LSTM(dim_embed, dim_lstm, num_layers=num_layers)
        self.fn = nn.Linear(dim_lstm, dim_out)
