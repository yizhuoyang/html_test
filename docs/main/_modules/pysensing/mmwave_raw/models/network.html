[docs]class Range_Fourier_Net(nn.Module):
    r"""
    The Range_Fourier_Net is a neural network module designed to apply a Fourier transformation to range-domain radar data. By utilizing complex-valued linear layers (CplxLinear), the network transforms raw range data into the frequency domain. The weights of the linear layer are pre-initialized with Fourier basis vectors to perform the discrete Fourier transform (DFT). This transformation is essential for extracting frequency components from the time-domain signals, which enhances feature representation for downstream tasks like classification.

    Reference: 
        https://arxiv.org/pdf/2111.03976
    """
    def __init__(self):
        super(Range_Fourier_Net, self).__init__()
        self.range_nn = CplxLinear(256, 256, bias = False)
        range_weights = np.zeros((256, 256), dtype = np.complex64)
        for j in range(0, 256):
            for h in range(0, 256):
                range_weights[h][j] = np.exp(-1j * 2 * np.pi *(j*h/256))
        range_weights = TensorToCplx()(torch.view_as_real(torch.from_numpy(range_weights)))
        self.range_nn.weight = CplxParameter(range_weights)

        x = self.range_nn(x)
        return x

    r"""
    The Doppler_Fourier_Net focuses on transforming Doppler-domain radar data into the frequency domain using complex-valued linear layers. Similar to the Range_Fourier_Net, it initializes its weights with Fourier basis vectors to perform the DFT. This transformation captures the velocity information of moving targets by analyzing the Doppler shifts, which is crucial for applications like motion detection and human action recognition.

    Reference: 
        https://arxiv.org/pdf/2111.03976
    """
    def __init__(self):
        super(Doppler_Fourier_Net, self).__init__()
        self.doppler_nn = CplxLinear(128, 128, bias = False)
        doppler_weights = np.zeros((128, 128), dtype=np.complex64)
        for j in range(0, 128):
            for h in range(0, 128):
                hh = h + 64
                if hh >= 128:
                    hh = hh - 128
                doppler_weights[h][j] = np.exp(-1j * 2 * np.pi * (j * hh / 128))
        doppler_weights = TensorToCplx()(torch.view_as_real(torch.from_numpy(doppler_weights)))
        self.doppler_nn.weight = CplxParameter(doppler_weights)

[docs]class AOA_Fourier_Net(nn.Module):
    r"""
    The AOA_Fourier_Net processes Angle-of-Arrival (AOA) data by transforming it into the frequency domain using complex-valued linear layers. It expands the 8-element AOA input into a 64-element frequency representation. This transformation is vital for accurately estimating the direction of incoming signals, which enhances spatial resolution in radar systems.

    Reference: 
        https://arxiv.org/pdf/2111.03976
    """
    def __init__(self):
        super(AOA_Fourier_Net, self).__init__()
        self.aoa_nn = CplxLinear(8, 64, bias = False)
        aoa_weights = np.zeros((64, 8), dtype=np.complex64)
        for j in range(8):
            for h in range(64):
                hh = h + 32
                if hh >= 64:
                    hh = hh - 64
                h_idx = h
                aoa_weights[h_idx][j] = np.exp(-1j * 2 * np.pi * (j *hh / 64))

        aoa_weights = TensorToCplx()(torch.view_as_real(torch.from_numpy(aoa_weights)))
        self.aoa_nn.weight = CplxParameter(aoa_weights)

[docs]class RT_2DCNN(nn.Module):
    r"""
    The RT_2DCNN (Range-Time 2D Convolutional Neural Network) processes range-time radar data. It first applies a Fourier transformation to the range data using the Range_Fourier_Net and then extracts features using 2D convolutional layers. 

    Input: 
        Raw 4D radar cube. Tensor size (batch, 10, 128, 12, 256) (batch size, Time, chirps, virtual antennas, virtual antenna per chirp)

    Output:
        Class probabilities over 12 action classes.

    Reference: 
        https://arxiv.org/pdf/2111.03976
    """
    def __init__(self, HAR=False):
        super(RT_2DCNN, self).__init__()
        self.HAR = HAR
        self.range_net = Range_Fourier_Net()
        self.cplx_transpose = CplxToCplx[torch.transpose]
        self.conv1 = nn.Conv2d(1, 4, 3)
        self.bn1 = nn.BatchNorm2d(4)
        self.conv2 = nn.Conv2d(4, 8, 3)
        self.bn2 = nn.BatchNorm2d(8)
        self.conv3 = nn.Conv2d(8, 16, 3)
        self.bn3 = nn.BatchNorm2d(16)
        self.maxpool = nn.MaxPool2d((1, 2), ceil_mode=True)

        T = 20 if HAR else 10

        # Calculate the output dimensions after convolutions and pooling
        def conv_output_size(size, kernel_size=3, stride=1, padding=0):
            return (size - kernel_size + 2 * padding) // stride + 1

        def pool_output_size(size, kernel_size=2):
            return math.ceil(size / kernel_size)

        H = T
        W = 256

        # Conv and Pool Layers
        H = conv_output_size(H)
        W = conv_output_size(W)
        H = pool_output_size(H, 1)
        W = pool_output_size(W)

        H = conv_output_size(H)
        W = conv_output_size(W)
        H = pool_output_size(H, 1)
        W = pool_output_size(W)

        H = conv_output_size(H)
        W = conv_output_size(W)
        H = pool_output_size(H, 1)
        W = pool_output_size(W)

        fc1_input_size = 16 * H * W

        self.fc_1 = nn.Linear(fc1_input_size, 512)
        self.fc_2 = nn.Linear(512, 128)
        self.fc_3 = nn.Linear(128, 5) if HAR else nn.Linear(128, 12)


[docs]    def forward(self, x):
        T = 20 if self.HAR else 10
        x = x[:, :, 0, 0, :]
        x = x.contiguous()
        x = self.range_net(x)
        x = CplxModulus()(x)
        x = x.view(-1, 1, T, 256)
        x = self.conv1(x)
        x = self.bn1(x)
        x = F.relu(x)
        x = self.maxpool(x)

        x = self.conv2(x)
        x = self.bn2(x)
        x = F.relu(x)
        x = self.maxpool(x)

        x = self.conv3(x)
        x = self.bn3(x)
        x = F.relu(x)
        x = self.maxpool(x)

        x = x.view(-1, self.fc_1.in_features)
        x = self.fc_1(x)
        x = F.relu(x)
        x = self.fc_2(x)
        x = F.relu(x)
        x = self.fc_3(x)
        return x
