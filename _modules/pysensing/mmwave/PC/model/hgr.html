[docs]class EVL_NN(nn.Module):
    r"""
    Implementation of human gesture recognition (hgr) following "M-Gesture: Person-Independent Real-Time In-Air Gesture Recognition Using Commodity Millimeter Wave Radar".

    Input:
        point clouds (tensor): Of shape (b, 50, 5). An aggregated sequence of point clouds using (1) sliding_window preprocessing, e.g. frame_together = 28 and sliding = 1 for M-Gesture; 
        (2) padding preprocessing, with npoints = 50 for M-Gesture dataset.

    Output:
        class label (tensor): Of shape (b, num_classes). The output class label in one-hot format.

    args:
        dataset (str): selected dataset name.

        num_classes (int): total number of output classes.

    Structure:
        Feature-engineering based module RPM, output 10 features for each frame of point cloud.
        1 * [2D CNN, MaxPooling]
        2 * [2D CNN, ReLU, 2D CNN]
        1 * [Linear head]
    """
    def __init__(self, dataset, num_classes):
        super(EVL_NN,self).__init__()
        self.dataset = dataset
        if dataset == 'M-Gesture':

            self.RPM_dim, self.frame = 10, 28
            self.C1 = nn.Sequential(
                nn.Conv2d(1, 32, (7, 7), stride=(2, 2), padding=(3, 3))
            )
            self.M = nn.MaxPool2d((3,3), stride=(2,2), padding=(0,1))
            self.E1 = nn.Sequential(
                nn.Conv2d(32, 32, (3, 3), stride=(1, 1), padding=(1, 1)),
                nn.ReLU(inplace=True),
                nn.Conv2d(32, 32, (3, 3), stride=(1, 1), padding=(1, 1)))
            self.E2 = nn.Sequential(
                nn.Conv2d(32, 128, (3, 3), stride=(1, 1), padding=(1, 1)),
                nn.ReLU(inplace=True),
                nn.Conv2d(128, 128, (3, 3), stride=(1, 1), padding=(1, 1))
                )
            self.F = nn.Sequential(
                nn.Linear(1792, 256),
                # nn.ReLU(),
            )
            
            self.classifier =  nn.Sequential(
                nn.Linear(256, num_classes),
                nn.Softmax(dim=-1))
        else:
            raise ValueError("Unsupported dataset.")


[docs]    def RPM_feat(self, x):
        # Overall

        # EC
        pc = x
        N_nonzero = torch.count_nonzero(pc, dim=-1)
        N_ec = torch.count_nonzero(N_nonzero, dim=-1)
        x_axis = pc[..., 0]
        y_axis = pc[..., 1]
        z_axis = pc[..., 2]
        V = pc[..., 3]*10
        I = pc[..., 4]
        V_ec = ((V*I).sum(-1)) / I.sum(-1)
        R = (x_axis.pow(2) + y_axis.pow(2) + z_axis.pow(2)).sqrt()
        R_ec = ((R*I).sum(-1)) / I.sum(-1)

        # CP
        idx_cp = V.gt(0)
        N_nonzero_cp = torch.count_nonzero(pc, dim=-1) * idx_cp
        N_cp = torch.count_nonzero(N_nonzero_cp, dim=-1)
        I_cp = I * idx_cp
        V_cp = ((V*I_cp).sum(-1)) / I_cp.sum(-1)
        R = (x_axis.pow(2) + y_axis.pow(2) + z_axis.pow(2)).sqrt()
        R_cp = ((R*I_cp).sum(-1)) / I_cp.sum(-1)

        # CF
        idx_cf = V.lt(0)
        N_nonzero_cf = torch.count_nonzero(pc, dim=-1) * idx_cf
        N_cf = torch.count_nonzero(N_nonzero_cf, dim=-1)
        I_cf = I * idx_cf
        V_cf = ((V*I_cf).sum(-1)) / I_cf.sum(-1)
        R = (x_axis.pow(2) + y_axis.pow(2) + z_axis.pow(2)).sqrt()
        R_cf = ((R*I_cf).sum(-1)) / I_cf.sum(-1)

        # alpha
        R[R==0] = 1000
        R_min, idx_R_min = torch.min(R, -1, keepdim = True)
        x_R_min = torch.gather(x_axis, -1, idx_R_min)
        y_R_min = torch.gather(y_axis, -1, idx_R_min)
        z_R_min = torch.gather(z_axis, -1, idx_R_min)

        alpha = (x_R_min/y_R_min)[...,0]

        RPM_map = torch.stack([N_ec, V_ec, R_ec, N_cp, V_cp, R_cp, N_cf, V_cf, R_cf, alpha], dim = -1)
        RPM_map = torch.nan_to_num(RPM_map, nan=0.0)

        return RPM_map




[docs]    def forward(self, x):
        x = self.RPM_feat(x)
        x = x.permute(0,2,1).unsqueeze(1)
        x = self.C1(x)
        x = self.M(x)
        x = self.E1(x)
        x = self.E2(x)
        x = torch.flatten(x, 1, -1)
        x = self.F(x)
        x = self.classifier(x)
        return x
