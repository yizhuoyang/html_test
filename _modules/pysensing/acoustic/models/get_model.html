[docs]def load_har_model(model_name: str = 'samcnn',
                   pretrained: bool = False,
                   progress: bool   = True):
    r'''
        Load the model for HAR task.

        Args:
            model_name(str):The model name. Default: 'samcnn'. Choose from 'samcnn','audiocnn'.
            pretrined (bool): If True, download the pretrained weights and load the weights to the model.
            progress (bool): If True, display the progress bar for the download 

        Returns:
            model: the model for the HAR task.
    '''
    if model_name == 'samcnn':
        model = HAR_SAMCNN()
    elif model_name == 'audiocnn':
        model = HAR_AUDIOCNN()
    else:
        raise ValueError("Unsupported model. Please choose from 'samcnn' or 'audiocnn'")
    
    if pretrained:
        state_dict = load_state_dict_from_url(model_urls[model_name],progress=progress)
        model.load_state_dict(state_dict)
    return model





[docs]def load_hpe_model(model_name: str = 'speech2pose',
                   pretrained: bool = False,
                   progress: bool   = True,
                   task:str = 'subject2'):
    r'''
        Load the model for HPE task.

        Args:
            model_name(str):The model name. Default: 'speech2pose'. Choose from 'speech2pose','wipose'
            pretrined (bool): If True, download the pretrained weights and load the weights to the model. Only provides the weights trained on logmel for speech2pose and
            raw for wipose, both the networks only trained on single subject setting 
            progress (bool): If True, display the progress bar for the download
            task (str): Only works when pretrained is equal to True. Load model trained on specific subject. Choose from [subject1, subject2...subject8]

        Returns:
            model: the model for the HPE task.

        Note:
            Only the weights for single subject and classroom setting is provided
    '''
    if model_name == 'speech2pose':
        model = Speech2pose()
    elif model_name == 'wipose':
        model = Wipose_LSTM()
    else:
        raise ValueError("Unsupported model. Please choose from 'speech2pose' or 'wipose'")
    
    if pretrained:
        model_name = f"{model_name}_{task}"
        state_dict = load_state_dict_from_url(model_urls[model_name],progress=progress)
        model.load_state_dict(state_dict)

    return model



[docs]def load_ped_det_model(model_name: str = 'ped_cnn',
                   pretrained: bool = False,
                   progress: bool   = True,
                   task:str = 'ideloc_ori_rd'):
    r'''
        Load the model for HPE task.

        Args:
            model_name(str):The model name. Default: 'ped_cnn'. Choose from 'ped_cnn','ped_crnn'.
            pretrined (bool): If True, download the pretrained weights and load the weights to the model.  
            progress (bool): If True, display the progress bar for the download
            task (str): Only works when the input model is ped_crnn. Load the model for the task selected from {ide/loc/ideloc/accil}_{ori/ana}_{rd/cloth/shoe} 

        Returns:
            model: the model for the Pedstrian Detection task.

        Note:
            For ped_crnn, only the weights for {ide/loc/ideloc/accil}_{ori}_{rd/cloth/shoe} are provided
    '''
    if model_name == 'ped_cnn':
        model = PED_CNN()
        if pretrained:
            state_dict = load_state_dict_from_url(model_urls[model_name],progress=progress)
            model.load_state_dict(state_dict)

    elif model_name == 'ped_crnn':
        model = PED_CRNN(task=task)
        task_type, train_type, covariant_type = task.split('_')
        if pretrained and train_type=='ori':
            model_name = f"{model_name}_{task}"
            state_dict = load_state_dict_from_url(model_urls[model_name],progress=progress)
            model.load_state_dict(state_dict['models']['model'])
    else:
        raise ValueError("Unsupported model. Please choose from 'ped_cnn' or 'ped_crnn'")

    return model
