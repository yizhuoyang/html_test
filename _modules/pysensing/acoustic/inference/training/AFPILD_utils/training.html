def train_main(config):
    # Do full cross-validation in a single thread
    verbosity = 2
    logger = get_logger('train', verbosity=verbosity)
    train_msg = msg_box("TRAIN: " + config.run_args.task)
    logger.debug(train_msg)

    # Setup GPU device if available, move model into configured device
    device, device_ids = prepare_device(config['n_gpu'])
    # Datasets
    train_datasets = dict()
    valid_datasets = dict()

    # Train dataset
    keys = ['datasets', 'train']
    for name in get_by_path(config, keys):
        train_datasets[name] = config.init_obj([*keys, name], 'data_loaders')
    # print(train_datasets)
    # With random splitting from the training dataset
    valid_exist = False  # Valid dataset

    # Data loaders
    train_data_loaders = dict()
    valid_data_loaders = dict()
    # Train dataloader
    keys = ['data_loaders', 'train']
    for name in get_by_path(config, keys):
        # Concat dataset
        if get_by_path(config, keys)[name]['type'] == "MultiDatasetDataLoader":
            train_data_loaders[name] = config.init_obj([*keys, name], 'data_loaders', train_datasets)
        else:
            dataset = train_datasets[name]
            train_data_loaders[name] = config.init_obj([*keys, name], 'data_loaders', dataset)

        if not valid_exist:
            valid_data_loaders[name] = train_data_loaders[name].valid_loader
    # print(train_data_loaders)
    # Models
    models_afpild = dict()
    logger_model = get_logger('model', verbosity=1)
    # print(config['models'])
    for name in config['models_afpild']:
        model = config.init_obj(['models_afpild', name], 'models_afpild')
        logger_model.info(model)
        logger.info(model)
        model = model.to(device)
        if len(device_ids) > 1:
            model = torch.nn.DataParallel(model, device_ids=device_ids)
        models_afpild[name] = model

    # Optimizers
    optimizers = dict()
    for name in config['optimizers']:
        trainable_params = filter(lambda p: p.requires_grad, models_afpild[name].parameters())
        optimizers[name] = config.init_obj(['optimizers', name], torch.optim, trainable_params)

    # Losses and metrics
    losses = dict()
    for name in config['losses']:
        kwargs = {}
        losses[name] = config.init_obj(['losses', name], module_loss, **kwargs)

    # Metrics
    metrics_iter = [getattr(module_metric, met) for met in config['metrics']['per_iteration']]
    metrics_epoch = [getattr(module_metric, met) for met in config['metrics']['per_epoch']]

    # Unchanged objects in each fold
    torch_args = {
        'datasets': {'train': train_datasets, 'valid': valid_datasets},
        'losses': losses,
        'metrics': {'iter': metrics_iter, 'epoch': metrics_epoch}
    }

    # Learning rate schedulers
    lr_schedulers = dict()
    for name in config['lr_schedulers']:
        lr_schedulers[name] = config.init_obj(['lr_schedulers', name],
                                              torch.optim.lr_scheduler, optimizers[name])

    # Update objects for each fold
    update_args = {
        'data_loaders': {'train': train_data_loaders, 'valid': valid_data_loaders},
        'models_afpild': models_afpild,
        'optimizers': optimizers,
        'lr_schedulers': lr_schedulers
    }
    torch_args.update(update_args)

    # Train entrance
    trainer = config.init_obj(['trainer'], 'trainers', torch_args, config.save_dir, config.resume, device)
    log_best = trainer.train()

    msg = msg_box("result")
    logger.info(f"{msg}\n{log_best}")

    config_file="pysensing/acoustic/inference/training/AFPILD_utils/configs/afpild_spec_gcc_fusion.json",
    root_dir = None,
    run_id=None,
    device=None,
    resume=None,
    mode='train',
    log_name=None,
    task='accil_ori_cloth',
    fold_idx=None,
    num_workers=None,
    lr=None,
    bs=None,
    tp=None,
    epochs=None,
    dataset_dir=None,
):
    """
    Function to train the AFPILD model.

    Args:
        config_file (str): Path to the JSON configuration file. 
        root_dir (str): Root directory for model and result saving.
        run_id (str, optional): Unique identifier for the current training run.
        device (str, optional): Device to run the model on (e.g., 'cpu', 'cuda').
        resume (str ): Path to a checkpoint to resume training from.
        mode (str, optional): Mode of operation. Options: ["train", "test"]. (Default: "train")
        log_name (str, optional): Name for the log file to record training progress and results.
        task (str, optional): Specific task to train on. (Default: "accil_ori_cloth")
        fold_idx (int, optional): Index of the fold for cross-validation, if applicable.
        num_workers (int, optional): Number of worker processes for data loading.
        lr (float, optional): Learning rate for the optimizer.
        bs (int, optional): Batch size for training.
        tp (str, optional): Training parameter (purpose to be specified).
        epochs (int, optional): Number of epochs to train for.
        dataset_dir (str): Directory containing the dataset.

    Returns:
        None

    Note:
        This function is based on AFPILD official repo: https://github.com/Cratial/AFPILD-CRNN

    Example:
        >>> afpild_train(config_file='custom_config.json', 
        ...             root_dir='./data/AFPILD',
        ...             task='accil_ana_shoe',
        ...             epochs=1,
        ...             num_workers=4,
        ...             dataset_dir='./data/AFPILD/'
        ...             root_dir='/path/to/project',
        ...             dataset_dir='/path/to/dataset')
    """

    # Create a parser
    parser = argparse.ArgumentParser(description='training')
    run_args = parser.add_argument_group('run_args')
    run_args.add_argument('-c', '--config', default=config_file, type=str)
    run_args.add_argument('--run_id', default=run_id or f'{strftime("%Y%m%d%H%M%S", gmtime())}', type=str)
    run_args.add_argument('-d', '--device', default=device, type=str)
    run_args.add_argument('-r', '--resume', default=resume, type=str)
    run_args.add_argument('--mode', default=mode, type=str)
    run_args.add_argument('--log_name', default=log_name, type=str)
    run_args.add_argument('-t', '--task', default=task, type=str)

    mod_args_group = parser.add_argument_group('mod_args')
    CustomArgs = collections.namedtuple('CustomArgs', "flags type target")
    options = [
        CustomArgs(['--root_dir'], type=str, target="root_dir"),
        CustomArgs(['--fold_idx'], type=int, target="trainer;fold_idx"),
        # fold_idx > 0 means multiprocessing is enabled
        CustomArgs(['--num_workers'], type=int,
                   target="data_loaders;train;data;kwargs;dataLoader_kwargs;num_workers"),
        CustomArgs(['--lr', '--learning_rate'], type=float, target="optimizers;model;kargs;lr"),
        CustomArgs(['--bs', '--batch_size'], type=int,
                   target="data_loaders;train;data;args;dataLoader_kwargs;batch_size"),
        CustomArgs(['--tp', '--transform_p'], type=float, target="datasets;train;data;kwargs;transform_p"),
        CustomArgs(['--epochs'], type=int, target=["trainer;kwargs;epochs", "trainer;kwargs;save_period"]),
        CustomArgs(['--dataset_dir'], type=str, target=["datasets;train;data;kwargs;dataset_dir","datasets;valid;data;kwargs;dataset_dir","datasets;test;data;kwargs;dataset_dir"]),
    ]
    for opt in options:
        mod_args_group.add_argument(*opt.flags, default=None, type=opt.type)

    # Build the list of command-line arguments
    cli_args = []
    if config_file is not None:
        cli_args.extend(['--config', config_file])
    if run_id is not None:
        cli_args.extend(['--run_id', run_id])
    if device is not None:
        cli_args.extend(['--device', device])
    if resume is not None:
        cli_args.extend(['--resume', resume])
    if mode is not None:
        cli_args.extend(['--mode', mode])
    if log_name is not None:
        cli_args.extend(['--log_name', log_name])
    if task is not None:
        cli_args.extend(['--task', task])

    # Now add optional arguments
    if fold_idx is not None:
        cli_args.extend(['--fold_idx', str(fold_idx)])
    if num_workers is not None:
        cli_args.extend(['--num_workers', str(num_workers)])
    if lr is not None:
        cli_args.extend(['--lr', str(lr)])
    if bs is not None:
        cli_args.extend(['--bs', str(bs)])
    if tp is not None:
        cli_args.extend(['--tp', str(tp)])
    if epochs is not None:
        cli_args.extend(['--epochs', str(epochs)])
    if dataset_dir is not None:
        cli_args.extend(['--dataset_dir', str(dataset_dir)])
    if root_dir is not None:
        cli_args.extend(['--root_dir', str(root_dir)])

    # run_args_dict = parse_args_to_class(parser)
    # print(run_args_dict.config)
    # Now pass cli_args to ConfigParser.from_args
    cfg = ConfigParser.from_args(parser, options, args_list=cli_args)
    train_main(cfg)
