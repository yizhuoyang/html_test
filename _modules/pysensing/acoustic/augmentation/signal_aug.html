[docs]class add_noise(torch.nn.Module):
    r"""
    Add noise on the input audio data

    Args:
        noise (torch.Tensor): The noise added on the audio data (need to be the same size with the audio data)
        snr (torch.Tensor): Signal-to-noise ratios in dB
        lengths (torch.Tensor or None, optional): Valid lengths of signals 

    Example:
        >>> import pysensing.acoustic.augmentation as A
        >>> noise = torch.randn_like(waveform)
        >>> snr   = torch.tensor([0])
        >>> add_noise_trans   = A.signal_aug.add_noise(noise,snr)
        >>> noise_data        = add_noise_trans(waveform)

    Reference: 
        https://pytorch.org/audio/main/generated/torchaudio.transforms.AddNoise.html?highlight=add+noise#torchaudio.transforms.AddNoise
    """
    def __init__(
        self,
        noise: torch.Tensor, 
        snr: torch.Tensor, 
        lengths: Optional[torch.Tensor] = None
    ) -> None:
        
        super(add_noise, self).__init__()
        self.noise   = noise
        self.snr     = snr
        self.lengths = lengths


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            torch.Tensor: the input raw audio
        Returns:
            torch.Tensor: the audio with added noise
        """
        add_noise_result = torchaudio.functional.add_noise(
            waveform,
            noise = self.noise,
            snr   = self.snr,
            lengths = self.lengths
        )

        return add_noise_result

    


[docs]class add_atempo(torch.nn.Module):
    r"""
    Speed up or slow down the input audio. The input waveform is channel last

    Args:
        sample_rate (int): The sample rate of the input audio
        atempo (float): The desired audio tempo. Tempo must be in the [0.5, 100.0] range. 

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> add_echo_trans   = A.signal_aug.add_echo(sample_rate,0.5)
        >>> echo_data        = add_echo_trans(waveform)

    Reference: 
        https://pytorch.org/audio/main/generated/torchaudio.io.AudioEffector.html#torchaudio.io.AudioEffector
        https://ffmpeg.org/ffmpeg-filters.html#atempo
    """
    def __init__(
        self,
        sample_rate: int,
        atempo: float=1.0,
    )-> None:
        
        super(add_atempo, self).__init__()
        self.sample_rate    = sample_rate
        self.transformation = torchaudio.io.AudioEffector(
            effect = f"atempo={atempo}",
        )


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            torch.Tensor: the input raw audio
        Returns:
            torch.Tensor: the audio with speed adjustment
        """
        add_atempo_result = self.transformation.apply(waveform,self.sample_rate)

        return add_atempo_result




[docs]class add_echo(torch.nn.Module):
    r"""
    Apply echo to the input audio. The input waveform is channel last

    Args:
        sample_rate (int): The sample rate of the input audio
        in_gain (float): Set input gain of reflected signal. The defalt value is 0.6 
        out_gain(float): Set output gain of reflected signal. Default is 0.3. 
        delays(List[int]): Set list of time intervals in milliseconds between original signal and reflections. Allowed range for each delay is (0 - 90000.0]. Default is 1000. 
        decays(List[float]): Set list of loudness of reflected signals. Allowed range for each decay is (0 - 1.0]. Default is 0.5. 

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> add_echo_trans   = A.signal_aug.add_echo(sample_rate,in_gain=0.6,out_gain=0.3,delays=[1000],decays=[0.5])
        >>> echo_data        = add_echo_trans(waveform)

    Reference: 
        https://pytorch.org/audio/main/tutorials/effector_tutorial.html#sphx-glr-tutorials-effector-tutorial-py
        https://ffmpeg.org/ffmpeg-filters.html#aecho
    """
    def __init__(
        self,
        sample_rate: int,
        in_gain: float=0.6,
        out_gain: float=0.3,
        delays:  List[int]=[1000],
        decays: List[float]=[0.5],
    )-> None:
        
        super(add_echo, self).__init__()
        self.sample_rate    = sample_rate
        self.transformation = torchaudio.io.AudioEffector(
            effect = f"aecho={in_gain}:{out_gain}:{'|'.join(map(str,delays))}:{'|'.join(map(str, decays))}",
        )


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            torch.Tensor: the input raw audio
        Returns:
            torch.Tensor: the audio with added echo
        """
        add_echo_result = self.transformation.apply(waveform,self.sample_rate)

        return add_echo_result

    


[docs]class add_chorus(torch.nn.Module):
    r"""
    Apply chorus effect to the input audio. The input waveform is channel last.

    The chorus effect simulates the sound of multiple voices from a single vocal track by adding delayed and modulated versions of the original signal.

    Args:
        sample_rate (int): The sample rate of the input audio
        delays(List[int]): This parameter specifies the delay times for each chorus voice. Typical delay values are around 40ms to 60ms. Allowed range for each delay is (0 - 90000.0].
        decays(List[float]): This parameter controls the decay of each chorus voice. It determines how quickly each chorus voice decays over time. Allowed range for each decay is (0 - 1.0]
        speeds(List[float]): This parameter controls the modulation speed of each chorus voice. Range is (0 - 1.0]
        depths(List[float]): Determines the modulation depth for each voice, typically larger than 1
        in_gain (float): Set input gain. This parameter adjusts the level of the input signal before applying the chorus effect. The defalt value is 0.4 
        out_gain(float): Set output gain. This parameter adjusts the level of the output signal after applying the chorus effect. Default is 0.4. 

    Example
        >>> import pysensing.acoustic.augmentation as A
        >>> add_chorus_trans   = A.signal_aug.add_chorus(sample_rate,in_gain=0.6,out_gain=0.3,delays=[1000],decays=[0.5],speeds=[0.25],depths=[2.0])
        >>> chorus_data        = add_chorus_trans(waveform)

    Reference: 
        https://pytorch.org/audio/main/tutorials/effector_tutorial.html#sphx-glr-tutorials-effector-tutorial-py
        https://ffmpeg.org/ffmpeg-filters.html#chorus
    """
    def __init__(
        self,
        sample_rate: int,
        delays: List[int],
        decays: List[float],
        speeds: List[float],
        depths: List[float],
        in_gain: float=0.4,
        out_gain: float=0.4,
    )-> None:
        
        super(add_chorus, self).__init__()
        self.sample_rate    = sample_rate
        self.transformation = torchaudio.io.AudioEffector(
            effect = f"chorus={in_gain}:{out_gain}:{'|'.join(map(str,delays))}:{'|'.join(map(str,decays))}:{'|'.join(map(str,speeds))}:{'|'.join(map(str,depths))}",
        )


[docs]    def forward(self, waveform: Tensor) -> Tensor:
        r"""
        Args:
            torch.Tensor: the input raw audio
        Returns:
            torch.Tensor: the audio with added chorus effect
        """
        add_chorus_result = self.transformation.apply(waveform,self.sample_rate)

        return add_chorus_result
