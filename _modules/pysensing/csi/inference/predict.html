[docs]def har_predict(csi: Tensor, dataset: str, model: nn.Module, device: device) -> Tensor:
    r'''
    Use the pretrained model to inference a frame of csi data.

    Args:
        csi: data to be inferenced, which is required to be the same size as the dataset provided.
        dataset: the dataset that csi belongs to
        model: the pretrained model

    Return:
        output: the predicted logits of classes

    Example:
        >>> csi = torch.zeros(250, 90)
        >>> model = load_har_model('UT_HAR', 'MLP')
        >>> device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        >>> model = load_pretrain(model, 'UT_HAR', 'MLP', device=device)
        >>> output = har_predict(csi, 'UT_HAR', model, device)
    '''
    dataset_shape = {'UT_HAR': [250, 90], 'NTUFi_HAR': [3, 114, 500], 'Widar3': [22, 20, 20]}
    if dataset not in dataset_shape:
        raise ValueError("Unsupported dataset. Please choose from 'UT_HAR', 'NTUFi_HAR', or 'Widar3'.")

    expected_shape = dataset_shape[dataset]
    if len(csi.shape) == len(expected_shape) + 1:
        for data in csi:
            if data.shape != tuple(expected_shape):
                raise ValueError(
                    f"The shape of one of the input csi data in the batch does not match the expected shape for dataset '{dataset}'.")
    elif len(csi.shape) == len(expected_shape):
        if csi.shape != tuple(expected_shape):
            raise ValueError(
                f"The shape of the input csi data does not match the expected shape for dataset '{dataset}'.")
        csi = csi.unsqueeze(0)  # 添加批次维度
    else:
        raise ValueError(f"The shape of the input csi data does not match the expected shape for dataset '{dataset}'.")
    if dataset != model.dataset:
        raise ValueError(f"The dataset of input csi '{dataset}' does not match the input dataset of model '{model.dataset}'.")

    model.to(device)
    csi = csi.to(device)
    with torch.no_grad():
        output = model(csi)
    output = output.to(device)
    output = output.type(torch.FloatTensor)

    return output
