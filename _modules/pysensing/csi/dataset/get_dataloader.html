[docs]def load_classification_dataset(dataset: str = 'UT_HAR', batch_size: int = 64) -> Tuple[DataLoader, DataLoader]:
    r'''
    Load HAR and HID dataset.

    Args:
        dataset: The name of the dataset that needs to be loaded. The choices are UT_HAR, NTUFi_HumanID, NTUFi_HAR and Widar3.

    Reture:
        dataloaders for train and val datasets.

    Example:
        >>>train_loader, test_loader = load_classification_dataset('UT_HAR')
    '''
    # root = 'pysensing/csi/data/'  # load from website, need to be revised
    new_dataset_dict = {'UT_HAR': 'UT-HAR', 'NTUFi_HumanID': 'NTU-Fi-HumanID', 'NTUFi_HAR': 'NTU-Fi_HAR', 'Widar3': 'Widardata'}
    new_dataset = new_dataset_dict[dataset]
    path_root = ' https://pysensing.oss-ap-southeast-1.aliyuncs.com/data/csi/'
    file_path = os.path.join(path_root, new_dataset + '.zip')
    save_root = 'data/csi/'
    if not os.path.exists(save_root):
        os.makedirs(save_root)
    print(os.path.abspath(save_root))
    if not os.path.exists(save_root + dataset):
        save_path = os.path.join(save_root, dataset + '.zip')
        response = requests.get(file_path, stream=True)
        print(f"Downloading dataset：{file_path}")
        if response.status_code == 200:
            with open(save_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
        else:
            raise ValueError("Download failed.")
        with zipfile.ZipFile(save_path, 'r') as zip_ref:
            zip_ref.extractall(save_root)
        os.remove(save_path)
        os.rename(save_root + new_dataset, save_root + dataset)


    if dataset == 'UT_HAR':
        print('using dataset: UT-HAR DATA')
        data = UT_HAR(save_root+ 'UT_HAR')
        train_set = torch.utils.data.TensorDataset(data['X_train'], data['y_train'])
        test_set = torch.utils.data.TensorDataset(torch.cat((data['X_val'], data['X_test']), 0),
                                                  torch.cat((data['y_val'], data['y_test']), 0))
        train_loader = torch.utils.data.DataLoader(train_set, batch_size=batch_size, shuffle=True,
                                                   drop_last=True)  # drop_last=True
        test_loader = torch.utils.data.DataLoader(test_set, batch_size=batch_size, shuffle=False)

    elif dataset == 'NTUFi_HumanID':
        print('using dataset: NTU-Fi-HumanID')

        train_set = NTUFi_HumanID(save_root + 'NTUFi_HumanID/train_amp/')
        test_set = NTUFi_HumanID(save_root + 'NTUFi_HumanID/test_amp/')
        train_loader = torch.utils.data.DataLoader(train_set,
                                                   batch_size=batch_size, shuffle=True)
        test_loader = torch.utils.data.DataLoader(test_set,
                                                  batch_size=batch_size, shuffle=False)

    elif dataset == 'NTUFi_HAR':
        print('using dataset: NTUFi_HAR')

        train_set = NTUFi_HAR(save_root + 'NTUFi_HAR/train_amp/')
        test_set = NTUFi_HAR(save_root + 'NTUFi_HAR/test_amp/')
        train_loader = torch.utils.data.DataLoader(train_set,
                                                   batch_size=batch_size, shuffle=True)
        test_loader = torch.utils.data.DataLoader(test_set,
                                                  batch_size=batch_size, shuffle=False)

    elif dataset == 'Widar3':
        print('using dataset: Widar')
        train_set = Widar3(save_root + 'Widardata/train/')
        test_set = Widar3(save_root + 'Widardata/test/')
        train_loader = torch.utils.data.DataLoader(train_set, batch_size=batch_size,
                                                   shuffle=True)
        test_loader = torch.utils.data.DataLoader(test_set, batch_size=batch_size,
                                                  shuffle=False)
    else:
        raise ValueError("Unsupported dataset. Please choose from 'NTUFi_HAR', 'UT_HAR', or 'Widar3'.")

    return train_loader, test_loader




[docs]def load_hpe_dataset(dataset_name: str = 'MMFi',
                     protocol: str = 'protocol3',
                     split_to_use: str = 'random',
                     data_unit: str = 'frame',
                     batch_size: int = 32,
                     random_seed: Optional[int] = 0,
                     random_ratio: Optional[float] = 0.8,
                     manual_train_subjects: Optional[list] = None,
                     manual_val_subjects: Optional[list] = None,
                     manual_train_actions: Optional[list] = None,
                     manual_val_actions: Optional[list] = None,
                     ) -> Tuple[DataLoader, DataLoader]:
    r'''
        Load MM-Fi and Wi-Pose dataset with the specific settings.
        Reference: https://github.com/ybhbingo/MMFi_dataset

        Args:
            The parameters can be found in the MMFi website.

        Reture:
            dataloaders for train and val datasets

        Example:
            >>>train_loader, val_loader = load_hpe_dataset(dataset_name='MMFi', protocol='protocol1', split_to_use='random_split', random_seed=0, random_ratio=0.8, batch_size=32, data_unit='frame')
    '''
    if dataset_name != 'MMFi':
        new_dataset_dict = {'WiPose': 'Wi-Pose'}
        new_dataset = new_dataset_dict[dataset_name]
        path_root = ' https://pysensing.oss-ap-southeast-1.aliyuncs.com/data/csi/'
        file_path = os.path.join(path_root, new_dataset + '.zip')
        save_root = 'data/csi/'
        if not os.path.exists(save_root):
            os.makedirs(save_root)
        if not os.path.exists(save_root + dataset_name):
            save_path = os.path.join(save_root, dataset_name + '.zip')
            response = requests.get(file_path, stream=True)
            print(f"Downloading dataset：{file_path}")
            if response.status_code == 200:
                with open(save_path, 'wb') as f:
                    for chunk in response.iter_content(chunk_size=8192):
                        f.write(chunk)
            else:
                raise ValueError("Download failed.")
            with zipfile.ZipFile(save_path, 'r') as zip_ref:
                zip_ref.extractall(save_root)
            os.remove(save_path)
            os.rename(save_root + new_dataset, save_root + dataset_name)

    if dataset_name == 'MMFi':
        if not os.path.exists('data/csi/MMFi_Dataset'):
            print('Need to download MMFi dataset from website, and put it into the data/csi/MMFi_Dataset folder')
            raise FileNotFoundError
        root = 'data/csi/MMFi_Dataset'  # load from website
        # root = '/data2/msc/zyj/MMFi_Dataset'
        database = MetaFi_Database(root)
        config_dataset = decode_settings(protocol, split_to_use, random_seed,
                                         random_ratio, manual_train_subjects, manual_val_subjects,
                                         manual_train_actions, manual_val_actions)
        train_dataset = MetaFi_Dataset(database, data_unit, **config_dataset['train_dataset'])
        val_dataset = MetaFi_Dataset(database, data_unit, **config_dataset['val_dataset'])
        train_loader = DataLoader(train_dataset, batch_size=batch_size,
                                  shuffle=True, drop_last=True)
        val_loader = DataLoader(val_dataset, batch_size=batch_size,
                                shuffle=False, drop_last=False)
    elif dataset_name == 'WiPose':
        # root = 'pysensing/csi/data/Wi-Pose'
        root = 'data/csi/WiPose'
        train_dataset = WiPose(root, istrain=True)
        val_dataset = WiPose(root, istrain=False)
        train_loader = DataLoader(train_dataset, batch_size=batch_size,
                                  shuffle=True, drop_last=True)
        val_loader = DataLoader(val_dataset, batch_size=batch_size,
                                shuffle=False, drop_last=False)
    else:
        raise ValueError('Unsupported dataset. Please choose from MMFi or WiPose.')
    return train_loader, val_loader



def load_recon_dataset(dataset_name: str = 'HandFi', batch_size: int = 64, return_train: Optional[bool] = False):
    r'''
    Load reconstruction dataset.

    Args:
        dataset: The name of the dataset that needs to be loaded. The choice is HandFi.
        return_train: Return train set or not

    Reture:
        dataloaders for train and val datasets.

    Example:
        >>>test_loader = load_recon_dataset('HandFi', return_train=False)
    '''
     
     
    path_root = 'https://pysensing.oss-ap-southeast-1.aliyuncs.com/data/csi/'
    file_path = os.path.join(path_root, dataset_name + '.zip')
    save_root = 'data/csi/'
    print(os.path.abspath(save_root))
    if not os.path.exists(save_root):
        os.makedirs(save_root)
    if not os.path.exists(save_root + dataset_name):
        save_path = os.path.join(save_root, dataset_name + '.zip')
        response = requests.get(file_path, stream=True)
        print(f"Downloading dataset：{file_path}")
        if response.status_code == 200:
            with open(save_path, 'wb') as f:
                for chunk in response.iter_content(chunk_size=8192):
                    f.write(chunk)
        else:
            raise ValueError("Download failed.")
        with zipfile.ZipFile(save_path, 'r') as zip_ref:
            zip_ref.extractall(save_root)
        os.remove(save_path)
     
    if dataset_name == 'HandFi':
        print('using dataset: HandFi')
        if return_train:
            train_csi, test_csi, train_joints, test_joints, train_image, test_image = HandFi(save_root + 'HandFi', return_train=True)
            train_set = cusDataset(train_csi, train_image, train_joints)
            test_set = cusDataset(test_csi, test_image, test_joints)
            train_loader = DataLoader(train_set, batch_size=batch_size, shuffle=True)
            test_loader = DataLoader(test_set, batch_size=batch_size, shuffle=False)
            return train_loader, test_loader
        else:
            csi_data, joints_data, images_data = HandFi(save_root + 'HandFi', return_train=False)
            dataset = cusDataset(csi_data, images_data, joints_data)
            test_loader = DataLoader(dataset, batch_size=batch_size, shuffle=False)
            return test_loader
    else:
        raise ValueError('Unsupported dataset. Please choose from HandFi.')
