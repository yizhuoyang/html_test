[docs]def UT_HAR(root_dir):
    r'''
    UT-HAR dataset.  https://ieeexplore.ieee.org/document/8067693
    The dataset is collected by the WiFi CSI tool in the University of Texas at Austin.
    The CSI size is 250x90, and the dataset contains 7 activities.
    classes : lie down, fall, walk, pickup, run, sit down, stand up
    train number : 3977
    test number : 996

    Args: 
        root_dir (string): Directory with all the data and label.

    Returns: 
        WiFi_data (dict): {'train_data':Tensor, 'train_label':Tensor, 'test_data':Tensor, 'test_label':Tensor}
    '''
    a = root_dir+'/data/*.csv'
    data_list = glob.glob(root_dir+'/data/*.csv')
    label_list = glob.glob(root_dir+'/label/*.csv')
    WiFi_data = {}
    for data_dir in data_list:
        data_name = data_dir.split('/')[-1].split('.')[0]
        with open(data_dir, 'rb') as f:
            data = np.load(f)
            data = data.reshape(len(data),250,90)
            data_norm = (data - np.min(data)) / (np.max(data) - np.min(data))
        WiFi_data[data_name] = torch.Tensor(data_norm)
    for label_dir in label_list:
        label_name = label_dir.split('/')[-1].split('.')[0]
        with open(label_dir, 'rb') as f:
            label = np.load(f)
        WiFi_data[label_name] = torch.Tensor(label)
    return WiFi_data

[docs]class NTUFi_HAR(Dataset):
    r'''
    NTU-Fi dataset for NTU-HAR. https://ieeexplore.ieee.org/document/9667414
    CSI size: 3x114x500
    NTU-HAR classes: 6 (box, circle, clean, fall, run, walk)
    train number: 936
    test number: 264

    Args: 
            root_dir (string): Directory with all the images.
            modal (CSIamp/CSIphase): CSI data modal
            transform (callable, optional): Optional transform to be applied

    Returns: 
            x (Tensor): CSI data, y (int): label
    '''

    def __init__(self, root_dir, modal='CSIamp', transform=None):
        
        self.root_dir = root_dir
        self.modal=modal
        self.transform = transform
        self.data_list = glob.glob(root_dir+'/*/*.mat')
        self.folder = glob.glob(root_dir+'/*/')
        self.category = {self.folder[i].split('/')[-2]:i for i in range(len(self.folder))}

    def __len__(self):
        return len(self.data_list)

    def __getitem__(self, idx):
        if torch.is_tensor(idx):
            idx = idx.tolist()
            
        sample_dir = self.data_list[idx]
        y = self.category[sample_dir.split('/')[-2]]
        x = sio.loadmat(sample_dir)[self.modal]
        
        # normalize
        x = (x - 42.3199)/4.9802
        
        # sampling: 2000 -> 500
        x = x[:,::4]
        x = x.reshape(3, 114, 500)
        
        if self.transform:
            x = self.transform(x)
        
        x = torch.FloatTensor(x)

        return x,y

    r'''
    WIDAR3 dataset.  https://ieeexplore.ieee.org/document/9516988
    BVP size: 22x20x20
    number of classes: 22
    classes :
    Push&Pull, Sweep, Clap, Slide, Draw-N(H), Draw-O(H),Draw-Rectangle(H),
    Draw-Triangle(H), Draw-Zigzag(H), Draw-Zigzag(V), Draw-N(V), Draw-O(V), Draw-1,
    Draw-2, Draw-3, Draw-4, Draw-5, Draw-6, Draw-7, Draw-8, Draw-9, Draw-10
    train number : 34926
    test number : 8726

    Args: 
        root_dir (string): Directory with all the data.

    Returns: 
        x (Tensor): BVP data, y (int): label

    '''
    def __init__(self,root_dir):
        self.root_dir = root_dir
        self.data_list = glob.glob(root_dir+'/*/*.csv')
        self.folder = glob.glob(root_dir+'/*/')
        self.category = {self.folder[i].split('/')[-2]:i for i in range(len(self.folder))}
        
    def __len__(self):
        return len(self.data_list)

    def __getitem__(self, idx):
        if torch.is_tensor(idx):
            idx = idx.tolist()
            
        sample_dir = self.data_list[idx]
        y = self.category[sample_dir.split('/')[-2]]
        x = np.genfromtxt(sample_dir, delimiter=',')
        
        # normalize
        x = (x - 0.0025)/0.0119
        
        # reshape: 22,400 -> 22,20,20
        x = x.reshape(22,20,20)
        # interpolate from 20x20 to 32x32
        # x = self.reshape(x)
        x = torch.FloatTensor(x)

        return x,y
