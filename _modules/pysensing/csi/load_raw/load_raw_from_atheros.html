def bit_convert(data, maxbit):
    if data & (1 << (maxbit - 1)):
        data -= (1 << maxbit)
    return data


def getCSIdata(recBuff):
    csiBuff = []
    cstStatus = csi_struct()
    # print recBuff
    # 0-24 status, 25-1734 csi(1710bytes include 2bytes payload_len), 1040bytes payload, 2bytes buf_len
    a = recBuff[0:25]
    status = struct.unpack('>Q2H11BH', recBuff[0:25])
    cstStatus.tstamp = status[0]
    cstStatus.csi_len = status[1]
    cstStatus.channel = status[2]
    cstStatus.phyerr = status[3]
    cstStatus.noise = status[4]
    cstStatus.rate = status[5]
    cstStatus.chanBW = status[6]
    cstStatus.num_tones = status[7]
    cstStatus.nr = status[8]
    cstStatus.nc = status[9]
    cstStatus.rssi = status[10]
    cstStatus.rssi_0 = status[11]
    cstStatus.rssi_1 = status[12]
    cstStatus.rssi_2 = status[13]
    cstStatus.payload_len = status[14]
    # pprint (vars(cstStatus))

    CSI_ST_LEN = 25
    start_len = cstStatus.csi_len
    csiBuff = struct.unpack('=%dB' % start_len, recBuff[CSI_ST_LEN:CSI_ST_LEN + start_len])
    # print len(csiBuff)
    bits_left = 16
    bitmask = (1 << 10) - 1
    idx = h_idx = 0
    h_data = csiBuff[idx]
    idx += 1
    h_data += (csiBuff[idx] << 8)
    idx += 1
    current_data = h_data & ((1 << 16) - 1)
    csi_matrix = np.zeros((cstStatus.nr, cstStatus.nc, cstStatus.num_tones), dtype=complex)

    for k in range(0, cstStatus.num_tones):
        for nc_idx in range(0, cstStatus.nc):
            for nr_idx in range(0, cstStatus.nr):
                if (bits_left - 10) < 0:
                    # print nr_idx,nc_idx,idx
                    h_data = csiBuff[idx]
                    idx += 1
                    h_data += (csiBuff[idx] << 8)
                    idx += 1
                    current_data += h_data << bits_left
                    bits_left += 16
                imag = current_data & bitmask
                imag = bit_convert(imag, 10)
                # printf("imag is: %d | ",imag);
                # csi_matrix[nr_idx][nc_idx][k].imag = imag

                bits_left -= 10;
                current_data = current_data >> 10;
                if (bits_left - 10) < 0:
                    h_data = csiBuff[idx]
                    idx += 1
                    h_data += (csiBuff[idx] << 8)
                    idx += 1
                    current_data += h_data << bits_left
                    bits_left += 16
                real = current_data & bitmask;
                real = bit_convert(real, 10);
                # printf("real is: %d |",real);
                # csi_matrix[nr_idx][nc_idx][k].real = real;
                csi_matrix[nr_idx][nc_idx][k] = imag * 1j + real
                # printf("real is: %d \n",csi_matrix[nr_idx][nc_idx][k].real);
                bits_left -= 10;
                current_data = current_data >> 10;
    # print idx
    # print csi_matrix
    return cstStatus, csi_matrix

[docs]def load_raw_msg(msg_csi: Tuple[str, Any, Any]) -> dict:
    r"""
    Load the amplitude and phase of WiFi CSI from atheros raw msg data.

    Args:
        msg_csi: message stored in the rosbag. Usually have the shape of (topic, message, timestamp).

    Return:
        Dictionary that contains the amplitude and phase of WiFi CSI.

    """
    topic, message, timestamp = msg_csi
    content_raw = message[1]
    content = content_raw[-1735:]
    csi_status, csi_raw_data = getCSIdata(content)

    amp = 20 * np.log10(np.abs(csi_raw_data))
    amp = amp[:, 0, :]
    phs = np.angle(csi_raw_data)
    phs = phs[:, 0, :]

    amp_record = amp.flatten().transpose()
    phs_record = phs.flatten().transpose()

    return {'CSIamp': amp_record, 'CSIphase': phs_record}



[docs]def load_raw_bin(msg_csi_dir: str) -> dict:
    r"""
    Load the amplitude and phase of WiFi CSI from atheros raw bin file.

    Args:
        msg_csi: message stored in the rosbag. Usually have the shape of (topic, message, timestamp).

    Return:
        Dictionary that contains the amplitude and phase of WiFi CSI.

    """
    with open(msg_csi_dir, "rb") as f:
        data = f.read()
    content_raw = data[1]
    content = content_raw[-1735:]
    csi_status, csi_raw_data = getCSIdata(content)

    amp = 20 * np.log10(np.abs(csi_raw_data))
    amp = amp[:, 0, :]
    phs = np.angle(csi_raw_data)
    phs = phs[:, 0, :]

    amp_record = amp.flatten().transpose()
    phs_record = phs.flatten().transpose()

    return {'CSIamp': amp_record, 'CSIphase': phs_record}
