[docs]def predict(data: list[dict], segmenter) -> list:
    r'''
    Predicts the FDI label for all points in each item of the provided data.

    Args:
        data (list[dict]): A list of dictionaries containing 'jaw_dir', 'case_dir', 'feature, 'category' and 'label' keys for each IOS mesh.
        segmenter (object): The segmentation model used for making predictions.

    Returns:
        A list containing predicted labels for each point in each mesh in the input data.
    
    Example:
        >>> model = DGCNN()
        >>> model = nn.DataParallel(model)
        >>> weight_path = "path/to/weight_of_DGCNN"
        >>> model.load_state_dict(torch.load(weight_path, weights_only=True, map_location="cuda"))
        >>> segmenter = Segmenter(model)
        >>> data = extract_data_from_root(dataset_path)
        >>> labels = predict(data, segmenter)   # labels' shape is (data_len, number_of_points, 33)
    '''
    # Each IOS requires predicting the FDI label for all points, so batch inference is not used.
    results = []
    for item in tqdm(data, total=len(data)):
        findices = generate_findices(item['feature'].shape[0], segmenter.num_pts)
        inf_res = segmenter.segment(item['feature'], item['category'], findices)    # all points' class probs in one mesh
        prediction = np.argmax(inf_res, axis=1).astype(int)                         # all points' predicted labels in one mesh
        results.append(segmenter.convert_clzz_to_label(prediction))
    return results




[docs]def visualization(mesh: trimesh.Trimesh, lbl: list) -> trimesh.Trimesh:
    r'''
    Visualizes a mesh by coloring each vertex based on the provided labels.

    Args:
        mesh (Trimesh): A Trimesh object loaded from a mesh file.
        lbl (list): A list containing FDI labels (0 indicates gum) for each vertex in the mesh.

    Returns:
        Trimesh: A Trimesh object where each vertex is colored according to the labels provided.

    Example:
        >>> mesh = trimesh.exchange.obj.load_obj(F)
        >>> lbl = [np.random.choice([0] + list(range(11, 19)) + list(range(21, 29)) + list(range(31, 39)) + list(range(41, 49))) for _ in range(len(mesh.vertices))]
        >>> colored_mesh = visualization(mesh, lbl)
        >>> colored_mesh.show()
    '''
    COLORS = create_colors()
    vertices_obj = [tuple(v) for v in mesh['vertices']]
    vertex_label_mapping = dict(zip(vertices_obj, lbl))
    
    mesh_obj = trimesh.exchange.load.load_kwargs(mesh)
    points_mesh = mesh_obj.vertices
    labels = [vertex_label_mapping[tuple(vertex)] for vertex in points_mesh]

    colored_mesh = mesh_obj.copy()
    for label in np.unique(labels):
        indices = np.where(labels == label)[0]
        colored_mesh.visual.vertex_colors[indices] = COLORS[label]

    return colored_mesh
