[docs]def to_dec(s: str) -> list[int]:
    r'''
    Convert a hexadecimal color string to a list of RGBA values.

    Args:
        s (str): A hexadecimal string representing a color (e.g., 'b1cbbb').

    Returns:
        A list containing four elements [R, G, B, A]. R, G, B are the decimal values of the color components, and A is set to 255 (fully opaque).
    '''
    return [int(s[:2], 16), int(s[2:4], 16), int(s[4:6], 16), 255]



[docs]def create_colors() -> dict[int, list[int]]:
    r'''
    Create a dictionary mapping FDI labels to RGBA color values for visualization purposes.

    Returns:
        Dict[int, list[int]]: A dictionary where the keys are FDI labels and the values are RGBA color lists. The labels include gum color (0), upper teeth (11-18, 21-28), and lower teeth (31-38, 41-48).
    '''
    # 20 high contrast colors
    ss = ['b1cbbb', 'e3eaa7', 'eea29a', 'c94c4c', '92a8d1', 'f7cac9', 'f7786b', '034f84',
          'b5e7a0', '80ced6', 'fefbd8', '618685', 'ffef96', 'f18973', 'c0ded9', '82b74b']
    colors = [[255, 251, 240, 255]]      # gum color
    for s in ss:
        colors.append(to_dec(s))

    u_id = [0, 11, 12, 13, 14, 15, 16, 17, 18, 21, 22, 23, 24, 25, 26, 27, 28]
    u_colors = dict()
    for i, k in enumerate(u_id):
        u_colors[k] = colors[i]
    
    l_id = [0, 31, 32, 33, 34, 35, 36, 37, 38, 41, 42, 43, 44, 45, 46, 47, 48]
    l_colors = dict()
    for i, k in enumerate(l_id):
        l_colors[k] = colors[i]

    return {**l_colors, **u_colors}




[docs]def generate_findices(face_numbers: int, n_points: int) -> np.ndarray:
    r'''
    Generate a list of face indices for selecting a subset of points from a mesh.

    This function is used to create indices for inference by selecting subsets of mesh faces in batches.
    It ensures that all points are included and, if necessary, repeated to fill the batches.

    Args:
        face_numbers (int): The total number of faces available in the mesh.
        n_points (int): The number of points to be selected in each batch.

    Returns:
        An array of indices representing the selected points from the mesh. The length of this array will be `max_batch * n_points`.
    '''
    final_list = []
    max_batch = math.ceil(face_numbers / n_points)
    all_pts = max_batch * n_points
    inf_pts = [0] * all_pts
    indices1 = np.arange(face_numbers, dtype=int)
    indices2 = np.random.choice(indices1, replace=False, size=all_pts - face_numbers).astype(int)
    inf_pts[:face_numbers] = indices1[:]
    inf_pts[face_numbers:] = indices2[:]
    for i in range(max_batch):
        final_list += inf_pts[i::max_batch]
    return np.array(final_list)
