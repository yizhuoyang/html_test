[docs]class Synchronizer:
    r'''
    Synchronize the Bluetooth data from various MACs/Devices at a same location.

    Note:
        - The input raw data record should be in either ``.txt`` or ``.csv`` format,
          where ``.csv`` is highly recommended.
        - Each line of the data file should be in the order of ``Time``, ``Mac address``,
          ``Bluetooth value``, which is also the default order of this class.
        - The header (description) is recommended to appear at the first row of the read file.
          Otherwise, the default order will be automatically used.

    Examples:
        >>> An example of the data file should be like the following format.

        >>> Timestamp,MAC,BLE
        >>> 2024-05-01 17:11:59,BB:BB:BB:BB:BB:BB,64
        >>> 2024-05-01 17:12:54,DD:DD:DD:DD:DD:DD,72
        >>> 2024-05-01 17:13:02,EE:EE:EE:EE:EE:EE,10
        >>> 2024-05-01 17:12:42,CC:CC:CC:CC:CC:CC,30
        >>> 2024-05-01 17:11:42,AA:AA:AA:AA:AA:AA,65
        >>> 2024-05-01 17:11:44,BB:BB:BB:BB:BB:BB,55
        >>> 2024-05-01 17:11:46,BB:BB:BB:BB:BB:BB,42
        >>> 2024-05-01 17:11:45,DD:DD:DD:DD:DD:DD,49
        >>> 2024-05-01 17:11:43,CC:CC:CC:CC:CC:CC,57
        >>> 2024-05-01 17:11:43,AA:AA:AA:AA:AA:AA,61
        >>> 2024-05-01 17:11:44,EE:EE:EE:EE:EE:EE,60
    '''
    def __init__(self):
        self.sync_level = None
        self.list_mac = []
        self.record_raw = []


[docs]    def update_keys(self, all_keys):
        for idx in range(len(all_keys)):
            if 'time' or 'date' in all_keys[idx].lower():
                self.key_time = all_keys[idx]
            elif 'mac' in all_keys[idx].lower():
                self.key_mac = all_keys[idx]
            elif 'ble' in all_keys[idx].lower():
                self.key_ble = all_keys[idx]
            else:
                pass



[docs]    def read_file(self, path_file, has_header=True, descriptors_order=['time', 'mac', 'ble']):
        r'''
        Read unsynchronized raw data file and sort the new record in the time ascending order.

        Args:
            path_file (:obj:`String`): The raw data file.
            has_header (:obj:`Boolean`): The flag to specify whether the head is . Default: ``True``.
            descriptors_order (:obj:`List`): The order of defualt value descriptions. Default: ``time``, ``mac``, ``ble``.
        '''
        file_type = path_file.split('.')[-1]

        record_raw = []
        if file_type == 'txt':
            with open(path_file, encoding='utf-8') as file_obj:
                lines_read = file_obj.readlines()
                for line_read in lines_read:
                    data_raw_tmp = eval(line_read)
                    record_raw.append(data_raw_tmp)
            record_raw = pd.DataFrame(record_raw)
        elif file_type == 'csv':
            if has_header:
                record_raw = pd.read_csv(path_file)
            else:
                record_raw = pd.read_csv(path_file, header=None)
        else:
            raise ValueError('Only txt or csv files are supported.')

        if has_header:
            self.update_keys(record_raw.keys())
        else:
            """
            If there is no explicit header in the original data file, we will manually assign the header keys to the 
            data file according to the 'descriptors_order' by default.
            """
            header = descriptors_order
            self.update_keys(header)

        time_format = type(record_raw.loc[0][self.key_time])
        if time_format is str:
            self.sync_level = 'second'
            for i in range(len(record_raw)):
                record_raw.loc[i, self.key_time] = time.mktime(
                    time.strptime(record_raw.loc[i, self.key_time], "%Y-%m-%d %H:%M:%S")
                )
        elif time_format is int:
            self.sync_level = 'second'
        elif time_format is float:
            self.sync_level = 'millisecond'
        else:
            raise ValueError('Time value should be one of (str, int, float), but invalid time format is detected.')

        self.record_raw = record_raw.sort_values(by=[self.key_time], ignore_index=True)
        self.list_mac = sorted(record_raw[self.key_mac].unique().tolist())



[docs]    def get_sync_data(self, path_file, has_header, interval):
        r'''
        Read the original data file and perform synchronization according to the interval.

        Args:
            path_file (:obj:`String`): The raw data file.
            has_header (:obj:`Boolean`): The flag to specify whether the head is . Default: ``True``.
            interval (:obj:`Int` or 'Float'): Time window size for synchronization. Unit: second.
        '''
        self.read_file(path_file, has_header)

        global_time_offset = self.record_raw.loc[0, self.key_time]
        self.record_raw[self.key_time] -= global_time_offset
        timestamps_all = self.record_raw[self.key_time]

        record_sync = pd.DataFrame(columns=self.list_mac+[self.key_time])

        time_start = 0.0
        time_end = time_start + interval
        sync_flag = False
        for time_tmp in timestamps_all:
            if not sync_flag and time_tmp >= time_start:
                sync_flag = True
                time_start = time_tmp
                time_end = time_tmp + interval

            if sync_flag:
                idx_selected = (time_start <= timestamps_all) & (timestamps_all < time_end)
                num_tmp = sum(idx_selected)
                record_selected = self.record_raw.loc[idx_selected].reset_index(drop=True)
                buf_tmp = pd.DataFrame(columns=self.list_mac, index=list(range(0,num_tmp)))
                for i in range(num_tmp):
                    buf_tmp.loc[i, record_selected.loc[i, self.key_mac]] = record_selected.loc[i, self.key_ble]
                buf_sync = pd.DataFrame(columns=self.list_mac+[self.key_time], index=[0])
                buf_sync[self.key_time] = time_start + global_time_offset
                buf_sync[self.list_mac] = buf_tmp.mean(axis=0)

                sync_flag = False
                time_start = time_end
                time_end = time_start + interval

                record_sync = pd.concat([record_sync, buf_sync], ignore_index=True)

        return record_sync
